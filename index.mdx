        document.addEventListener('DOMContentLoaded', function() {
          const searchBox = document.querySelector('.search-box');
          const searchInput = document.querySelector('.search-input');
          
          function triggerGlobalSearch() {
            // Try multiple methods to trigger Mintlify search
            // Method 1: Keyboard shortcut simulation
            const event = new KeyboardEvent('keydown', {
              key: 'k',
              code: 'KeyK',
              ctrlKey: true,
              metaKey: true,
              bubbles: true
            });
            document.dispatchEvent(event);
            
            // Method 2: Try to find and click Mintlify search button
            const searchButton = document.querySelector('[data-search-button]') || 
                                document.querySelector('.search-trigger') ||
                                document.querySelector('[aria-label*="search" i]');
            if (searchButton) {
              searchButton.click();
            }
            
            // Method 3: Try to focus on Mintlify search input
            setTimeout(() => {
              const mintlifySearch = document.querySelector('input[type="search"]') ||
                                   document.querySelector('input[placeholder*="search" i]');
              if (mintlifySearch && mintlifySearch !== searchInput) {
                mintlifySearch.focus();
              }
            }, 100);
          }
          
          if (searchBox) {
            searchBox.addEventListener('click', triggerGlobalSearch);
          }
          
          if (searchInput) {
            searchInput.addEventListener('click', triggerGlobalSearch);
            searchInput.addEventListener('focus', triggerGlobalSearch);
          }
        });
      `}
