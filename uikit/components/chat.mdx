### Configuration Options

<AccordionGroup>
  <Accordion title="Display Options">
    | Option | Description | Default |
    |--------|-------------|---------|
    | `showCreateButton` | Show new chat creation button | `true` |
    | `showSearchBar` | Display search functionality | `true` |
    | `showArchivedChats` | Include archived conversations | `false` |
    | `showOnlineStatus` | Display user online indicators | `true` |
  </Accordion>
  
  <Accordion title="Callback Functions">
    | Callback | Parameters | Description |
    |----------|------------|-------------|
    | `onChannelSelect` | `channelId: string` | Triggered when user selects a chat |
    | `onCreateChat` | None | Triggered when create chat is pressed |
    | `onSearchQuery` | `query: string` | Triggered on search input |
    | `onArchiveToggle` | `isArchived: boolean` | Triggered when toggling archived view |
  </Accordion>

    import com.amity.socialcloud.uikit.chat.room.AmityChatRoomActivity
    
    class ChatRoomActivity : AppCompatActivity() {
        
        companion object {
            private const val EXTRA_CHANNEL_ID = "channel_id"
            
            fun newIntent(context: Context, channelId: String): Intent {
                return Intent(context, ChatRoomActivity::class.java).apply {
                    putExtra(EXTRA_CHANNEL_ID, channelId)
                }
            }
        }
        
        override fun onCreate(savedInstanceState: Bundle?) {
            super.onCreate(savedInstanceState)
            
            val channelId = intent.getStringExtra(EXTRA_CHANNEL_ID) ?: return
            
            setupChatRoom(channelId)
        }
        
        private fun setupChatRoom(channelId: String) {
            val chatRoomFragment = AmityChatRoomFragment.Builder(channelId)
                .onMessageSent { message ->
                    Log.d("ChatRoom", "Message sent: ${message.messageId}")
                }
                .onMessageDeleted { messageId ->
                    Log.d("ChatRoom", "Message deleted: $messageId")
                }
                .enableMessageReactions(true)
                .enableFileSharing(true)
                .enableVoiceMessages(true)
                .build()
            
            supportFragmentManager.beginTransaction()
                .replace(android.R.id.content, chatRoomFragment)
                .commit()
        }
    }

    import React, { useState } from 'react';
    import { 
      AmityUiKitProvider,
      AmityUiKitChat,
      AmityUiKitChatRoom,
      AmityUiKitCreateChat 
    } from '@amityco/ui-kit';
    
    function ChatApp() {
      const [currentView, setCurrentView] = useState<'home' | 'room' | 'create'>('home');
      const [selectedChannelId, setSelectedChannelId] = useState<string>('');
      const [showCreateChat, setShowCreateChat] = useState(false);
      
      const handleChatSelect = (channelId: string) => {
        setSelectedChannelId(channelId);
        setCurrentView('room');
      };
      
      const handleChatCreated = (channelId: string) => {
        setSelectedChannelId(channelId);
        setCurrentView('room');
        setShowCreateChat(false);
      };
      
      const renderCurrentView = () => {
        switch (currentView) {
          case 'room':
            return (
              <AmityUiKitChatRoom
                channelId={selectedChannelId}
                onBack={() => setCurrentView('home')}
              />
            );
          case 'create':
            return (
              <AmityUiKitCreateChat
                onChatCreated={handleChatCreated}
                onCancel={() => setCurrentView('home')}
              />
            );
          default:
            return (
              <AmityUiKitChat
                onChannelSelect={handleChatSelect}
                onCreateChat={() => setCurrentView('create')}
              />
            );
        }
      };
      
      return (
        <AmityUiKitProvider
          apiKey="your-api-key"
          userId="user-id"
          displayName="User Name"
          apiRegion="us"
        >
          <div className="chat-app">
            {renderCurrentView()}
          </div>
        </AmityUiKitProvider>
      );
    }
    
    export default ChatApp;
