# Communities

Social+ UIKit community components provide comprehensive tools for creating, managing, and moderating communities within your application. These components handle everything from community setup to member management and content moderation.

## Overview

Communities enable users to create focused spaces for discussions, content sharing, and social interaction around specific topics or interests. The community system supports public and private communities with granular permission controls.

        } else if community.isPrivate && !community.isJoined {
            button.setTitle("Request to Join", for: .normal)
            button.backgroundColor = .systemOrange
        } else {
            button.setTitle("Join", for: .normal)
            button.backgroundColor = .systemBlue
        }
        
        button.addTarget(self, action: #selector(joinButtonTapped), for: .touchUpInside)
        return button
    }
    
    @objc private func joinButtonTapped() {
        if community.isJoined {
            leaveCommunity()
        } else {
            joinCommunity()
        }
    }

function CommunityJoinButton({ community, onToggle }) {
  const [isLoading, setIsLoading] = useState(false);

  const handleClick = async () => {
    setIsLoading(true);
    try {
      await onToggle();
    } finally {
      setIsLoading(false);
    }

    if (isLoading) return 'Loading...';
    if (community.isJoined) return 'Leave';
    if (community.isPrivate && !community.isJoined) return 'Request to Join';
    return 'Join';

    if (community.isJoined) return 'leave-button';

    return 'join-button';

    <button
      className={`community-action-button ${getButtonClass()}`}
      onClick={handleClick}
      disabled={isLoading}
    >
      {getButtonText()}
    </button>
  );

class CommunityMembersViewController: UIViewController {
    private let communityId: String
    private var members: [AmityCommunityMember] = []
    private let tableView = UITableView()
    private let searchController = UISearchController()
    
    init(communityId: String) {
        self.communityId = communityId
        super.init(nibName: nil, bundle: nil)
    }
    
    override func viewDidLoad() {
        super.viewDidLoad()
        setupUI()
        loadMembers()
    }
    
    private func setupUI() {
        title = "Members"
        
        // Setup search
        searchController.searchResultsUpdater = self
        searchController.obscuresBackgroundDuringPresentation = false
        searchController.searchBar.placeholder = "Search members"
        navigationItem.searchController = searchController
        
        // Setup table view
        tableView.delegate = self
        tableView.dataSource = self
        tableView.register(MemberTableViewCell.self, forCellReuseIdentifier: "MemberCell")
        
        view.addSubview(tableView)
        tableView.translatesAutoresizingMaskIntoConstraints = false
        NSLayoutConstraint.activate([
            tableView.topAnchor.constraint(equalTo: view.safeAreaLayoutGuide.topAnchor),
            tableView.leadingAnchor.constraint(equalTo: view.leadingAnchor),
            tableView.trailingAnchor.constraint(equalTo: view.trailingAnchor),
            tableView.bottomAnchor.constraint(equalTo: view.bottomAnchor)
        ])
        
        // Add member management options if user is moderator
        if currentUserCanModerate() {
            setupModeratorOptions()
        }
    }
    
    private func loadMembers() {
        AmityCommunityRepository().getMembers(communityId: communityId) { [weak self] result in
            DispatchQueue.main.async {
                switch result {
                case .success(let members):
                    self?.members = members
                    self?.tableView.reloadData()
                case .failure(let error):
                    self?.showError("Failed to load members: \(error.localizedDescription)")
                }
            }
        }
    }
    
    private func setupModeratorOptions() {
        let addMemberButton = UIBarButtonItem(
            barButtonSystemItem: .add,
            target: self,
            action: #selector(addMemberTapped)
        )
        navigationItem.rightBarButtonItem = addMemberButton
    }
    
    @objc private func addMemberTapped() {
        let userSearchViewController = UserSearchViewController()
        userSearchViewController.onUserSelected = { [weak self] user in
            self?.inviteUser(user)
        }
        present(userSearchViewController, animated: true)
    }
}

// MARK: - TableView DataSource & Delegate
extension CommunityMembersViewController: UITableViewDataSource, UITableViewDelegate {
    func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        return members.count
    }
    
    func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        let cell = tableView.dequeueReusableCell(withIdentifier: "MemberCell", for: indexPath) as! MemberTableViewCell
        let member = members[indexPath.row]
        
        cell.configure(
            avatar: member.user?.avatar?.fileUrl,
            name: member.user?.displayName ?? "Unknown",
            role: member.roles.first?.rawValue ?? "Member",
            joinDate: member.createdAt
        )
        
        // Add context menu for moderators
        if currentUserCanModerate() && member.userId != currentUserId {
            cell.contextMenuConfiguration = createMemberContextMenu(for: member)
        }
        
        return cell
    }
    
    func tableView(_ tableView: UITableView, didSelectRowAt indexPath: IndexPath) {
        tableView.deselectRow(at: indexPath, animated: true)
        let member = members[indexPath.row]
        
        // Navigate to user profile
        let userProfileViewController = AmityUserProfileViewController(userId: member.userId)
        navigationController?.pushViewController(userProfileViewController, animated: true)
    }
}

class CommunityMembersFragment : Fragment() {
    private lateinit var binding: FragmentCommunityMembersBinding
    private lateinit var membersAdapter: CommunityMembersAdapter
    private lateinit var communityRepository: AmityCommunityRepository
    private var communityId: String = ""
    private var members = mutableListOf<AmityCommunityMember>()

    override fun onCreateView(inflater: LayoutInflater, container: ViewGroup?, savedInstanceState: Bundle?): View {
        binding = FragmentCommunityMembersBinding.inflate(inflater, container, false)
        return binding.root
    }

    override fun onViewCreated(view: View, savedInstanceState: Bundle?) {
        super.onViewCreated(view, savedInstanceState)
        
        communityId = arguments?.getString(ARG_COMMUNITY_ID) ?: ""
        communityRepository = AmityCommunityRepository()
        
        setupRecyclerView()
        setupSearchView()
        loadMembers()
        
        if (currentUserCanModerate()) {
            setupModeratorOptions()
        }
    }

    private fun setupRecyclerView() {
        membersAdapter = CommunityMembersAdapter(
            onMemberClick = { member -> navigateToUserProfile(member.userId) },
            onMemberLongClick = { member -> showMemberOptions(member) }
        )
        
        binding.recyclerViewMembers.apply {
            adapter = membersAdapter
            layoutManager = LinearLayoutManager(requireContext())
            addItemDecoration(DividerItemDecoration(requireContext(), DividerItemDecoration.VERTICAL))
        }
    }

    private fun setupSearchView() {
        binding.searchView.setOnQueryTextListener(object : SearchView.OnQueryTextListener {
            override fun onQueryTextSubmit(query: String?): Boolean {
                filterMembers(query)
                return true
            }

            override fun onQueryTextChange(newText: String?): Boolean {
                filterMembers(newText)
                return true
            }
        })
    }

    private fun loadMembers() {
        communityRepository.getMembers(communityId)
            .subscribeOn(Schedulers.io())
            .observeOn(AndroidSchedulers.mainThread())
            .subscribe(
                { membersList ->
                    members.clear()
                    members.addAll(membersList)
                    membersAdapter.submitList(members.toList())
                },
                { error ->
                    showError("Failed to load members: ${error.message}")
                }
            )
    }

    private fun filterMembers(query: String?) {
        val filteredMembers = if (query.isNullOrBlank()) {
            members
        } else {
            members.filter { member ->
                member.user?.displayName?.contains(query, ignoreCase = true) == true
            }
        }
        membersAdapter.submitList(filteredMembers)
    }

    private fun setupModeratorOptions() {
        binding.fabAddMember.apply {
            visibility = View.VISIBLE
            setOnClickListener { showAddMemberDialog() }
        }
    }

    private fun showMemberOptions(member: AmityCommunityMember) {
        if (!currentUserCanModerate() || member.userId == getCurrentUserId()) return

        val options = arrayOf(
            "Promote to Moderator",
            "Remove from Community",
            "View Profile"
        )

        AlertDialog.Builder(requireContext())
            .setTitle("Member Options")
            .setItems(options) { _, which ->
                when (which) {
                    0 -> promoteToModerator(member)
                    1 -> removeMember(member)
                    2 -> navigateToUserProfile(member.userId)
                }
            }
            .show()
    }

    private fun promoteToModerator(member: AmityCommunityMember) {
        communityRepository.assignRole(communityId, member.userId, AmityCommunityRole.MODERATOR)
            .subscribeOn(Schedulers.io())
            .observeOn(AndroidSchedulers.mainThread())
            .subscribe(
                { 
                    showToast("${member.user?.displayName} promoted to moderator")
                    loadMembers() // Refresh list
                },
                { error -> showError("Failed to promote member: ${error.message}") }
            )
    }

    private fun removeMember(member: AmityCommunityMember) {
        AlertDialog.Builder(requireContext())
            .setTitle("Remove Member")
            .setMessage("Are you sure you want to remove ${member.user?.displayName} from this community?")
            .setPositiveButton("Remove") { _, _ ->
                communityRepository.removeMember(communityId, member.userId)
                    .subscribeOn(Schedulers.io())
                    .observeOn(AndroidSchedulers.mainThread())
                    .subscribe(
                        { 
                            showToast("Member removed successfully")
                            loadMembers()
                        },
                        { error -> showError("Failed to remove member: ${error.message}") }
                    )
            }
            .setNegativeButton("Cancel", null)
            .show()
    }

function CommunityMembers({ communityId, userRole }) {
  const [members, setMembers] = useState([]);
  const [filteredMembers, setFilteredMembers] = useState([]);
  const [searchQuery, setSearchQuery] = useState('');
  const [loading, setLoading] = useState(true);
  const [selectedMembers, setSelectedMembers] = useState([]);
  const [showBulkActions, setShowBulkActions] = useState(false);

  useEffect(() => {
    loadMembers();
  }, [communityId]);

  useEffect(() => {
    filterMembers();
  }, [members, searchQuery]);

  const loadMembers = async () => {
    try {
      const membersList = await getCommunityMembers(communityId);
      setMembers(membersList);
    } catch (error) {
      console.error('Failed to load members:', error);
    } finally {
      setLoading(false);
    }
  };

  const filterMembers = () => {
    if (!searchQuery) {
      setFilteredMembers(members);
    } else {
      setFilteredMembers(
        members.filter(member =>
          member.user.displayName.toLowerCase().includes(searchQuery.toLowerCase()) ||
          member.user.username?.toLowerCase().includes(searchQuery.toLowerCase())
        )
      );
    }
  };

  const handleMemberSelect = (memberId, isSelected) => {
    if (isSelected) {
      setSelectedMembers(prev => [...prev, memberId]);
    } else {
      setSelectedMembers(prev => prev.filter(id => id !== memberId));
    }
  };

  const handleBulkAction = async (action) => {
    try {
      switch (action) {
        case 'remove':
          await Promise.all(selectedMembers.map(memberId => 
            removeCommunityMember(communityId, memberId)
          ));
          break;
        case 'promote':
          await Promise.all(selectedMembers.map(memberId => 
            promoteMemberToModerator(communityId, memberId)
          ));
          break;
      }
      
      setSelectedMembers([]);
      setShowBulkActions(false);
      loadMembers();
    } catch (error) {
      console.error('Bulk action failed:', error);
    }
  };

  const canModerate = userRole === 'admin' || userRole === 'moderator';

  if (loading) return <MembersListSkeleton />;

  return (
    <div className="community-members">
      <div className="members-header">
        <h2>Members ({members.length})</h2>
        
        <div className="members-actions">
          <div className="search-box">
            <input
              type="text"
              placeholder="Search members..."
              value={searchQuery}
              onChange={(e) => setSearchQuery(e.target.value)}
            />
          </div>
          
          {canModerate && (
            <div className="moderator-actions">
              <button
                className="bulk-select-toggle"
                onClick={() => setShowBulkActions(!showBulkActions)}
              >
                {showBulkActions ? 'Cancel' : 'Select Multiple'}
              </button>
              
              <button className="invite-members">
                Invite Members
              </button>
            </div>
          )}
        </div>
      </div>

      {showBulkActions && selectedMembers.length > 0 && (
        <div className="bulk-actions">
          <span>{selectedMembers.length} selected</span>
          <div className="action-buttons">
            <button onClick={() => handleBulkAction('promote')}>
              Promote to Moderator
            </button>
            <button onClick={() => handleBulkAction('remove')} className="danger">
              Remove Members
            </button>
          </div>
        </div>
      )}

      <div className="members-list">
        {filteredMembers.map(member => (
          <MemberCard
            key={member.id}
            member={member}
            canModerate={canModerate}
            showCheckbox={showBulkActions}
            isSelected={selectedMembers.includes(member.id)}
            onSelect={(isSelected) => handleMemberSelect(member.id, isSelected)}
            onRoleChange={() => loadMembers()}
          />
        ))}
      </div>
    </div>
  );
}
