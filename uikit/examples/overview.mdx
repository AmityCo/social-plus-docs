### Quick Component Integration

<CodeGroup>
```javascript React
import React from 'react';
import { SocialFeed, PostCreator } from '@socialplus/uikit-react';

function SocialApp() {
  const handlePostCreated = (post) => {
    console.log('New post created:', post);
    // Refresh feed or add post optimistically
  };

  return (
    <div className="social-app">
      <PostCreator 
        onPostCreated={handlePostCreated}
        allowedMediaTypes={['image', 'video']}
        maxMediaSize={10 * 1024 * 1024} // 10MB
      />
      
      <SocialFeed
        communityId="general"
        enableRealTimeUpdates={true}
        onPostClick={(post) => console.log('Post clicked:', post)}
        customization={{
          theme: 'light',
          primaryColor: '#1054DE'
        }}
      />
    </div>
  );
}

export default SocialApp;
```

```swift iOS
import UIKit
import SocialPlusUIKit

class SocialViewController: UIViewController {
    
    override func viewDidLoad() {
        super.viewDidLoad()
        setupSocialFeed()
    }
    
    private func setupSocialFeed() {
        let socialFeed = SPUSocialFeedViewController(
            communityId: "general"
        )
        
        socialFeed.delegate = self
        socialFeed.customization = SPUCustomization(
            theme: .light,
            primaryColor: UIColor(hex: "#1054DE")
        )
        
        addChild(socialFeed)
        view.addSubview(socialFeed.view)
        socialFeed.didMove(toParent: self)
        
        // Setup constraints
        socialFeed.view.translatesAutoresizingMaskIntoConstraints = false
        NSLayoutConstraint.activate([
            socialFeed.view.topAnchor.constraint(equalTo: view.safeAreaLayoutGuide.topAnchor),
            socialFeed.view.leadingAnchor.constraint(equalTo: view.leadingAnchor),
            socialFeed.view.trailingAnchor.constraint(equalTo: view.trailingAnchor),
            socialFeed.view.bottomAnchor.constraint(equalTo: view.bottomAnchor)
        ])
    }
}

extension SocialViewController: SPUSocialFeedDelegate {
    func socialFeed(_ socialFeed: SPUSocialFeedViewController, didSelectPost post: SPUPost) {
        // Handle post selection
        print("Post selected: \(post.id)")
    }
}
```

```kotlin Android
import androidx.appcompat.app.AppCompatActivity
import android.os.Bundle
import com.socialplus.uikit.SocialFeedFragment
import com.socialplus.uikit.models.Customization
import com.socialplus.uikit.models.Theme

class MainActivity : AppCompatActivity(), SocialFeedFragment.EventListener {
    
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
        
        setupSocialFeed()
    }
    
    private fun setupSocialFeed() {
        val socialFeed = SocialFeedFragment.Builder()
            .communityId("general")
            .eventListener(this)
            .customization(
                Customization.Builder()
                    .theme(Theme.LIGHT)
                    .primaryColor("#1054DE")
                    .build()
            )
            .build()
            
        supportFragmentManager.beginTransaction()
            .replace(R.id.fragment_container, socialFeed)
            .commit()
    }
    
    override fun onPostClicked(postId: String) {
        // Handle post click
        println("Post clicked: $postId")
    }
}
```

```dart Flutter
import 'package:flutter/material.dart';
import 'package:socialplus_uikit/socialplus_uikit.dart';

class SocialApp extends StatefulWidget {
  @override
  _SocialAppState createState() => _SocialAppState();
}

class _SocialAppState extends State<SocialApp> {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Social Feed'),
        backgroundColor: Color(0xFF1054DE),
      ),
      body: Column(
        children: [
          PostCreator(
            onPostCreated: (post) {
              print('New post created: ${post.id}');
              // Handle post creation
            },
            allowedMediaTypes: [MediaType.image, MediaType.video],
            maxMediaSize: 10 * 1024 * 1024, // 10MB
          ),
          Expanded(
            child: SocialFeed(
              communityId: 'general',
              onPostTap: (post) {
                print('Post tapped: ${post.id}');
              },
              customization: Customization(
                theme: Theme.light,
                primaryColor: Color(0xFF1054DE),
              ),
            ),
          ),
        ],
      ),
    );
  }
}
```
