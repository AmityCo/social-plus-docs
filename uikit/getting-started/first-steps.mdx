# First Steps

Now that you have UIKit installed, let's get you up and running with authentication and your first social component. This guide will take you from a fresh installation to a working social feature in under 10 minutes.

<Info>
**Prerequisites**: Make sure you've completed the [installation guide](/uikit/getting-started/installation) for your platform before proceeding.
</Info>

## Step 1: Get Your API Credentials

Before you can use UIKit, you need to get your API key from the Social+ Console:

<Steps>
  <Step title="Access Social+ Console">
    Visit the [Social+ Console](https://console.socialplus.com) and log in to your account.
  </Step>
  
  <Step title="Navigate to Settings">
    Click **Settings** in the left sidebar, then select **Security**.
  </Step>
  
  <Step title="Copy Your Credentials">
    You'll need:
    - **API Key**: Found in the Keys section
    - **Region**: Your application's data center region (US, EU, or SG)
  </Step>
</Steps>

<Warning>
Keep your API key secure and never expose it in client-side code. For production applications, implement proper server-side authentication.
</Warning>

## Step 2: Initialize UIKit

Configure UIKit with your credentials in your application:

<Tabs>
  <Tab title="iOS (Swift)">
    **In your AppDelegate or SceneDelegate:**
    
    ```swift
    import AmityUIKit

    // In AppDelegate
    func application(_ application: UIApplication, didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?) -> Bool {
        
        AmityUIKitManager.setup(
            apiKey: "YOUR_API_KEY",
            region: .US // Change to .EU or .SG based on your region
        )
        
        return true
    }
    ```

    **Register a user:**
    ```swift
    AmityUIKitManager.registerDevice(withUserId: "user123") { success, error in
        if success {
            print("User registered successfully")
        } else {
            print("Registration failed: \(error?.localizedDescription ?? "Unknown error")")
        }
    }
    ```
  </Tab>

  <Tab title="Android (Kotlin)">
    **In your Application class:**
    
    ```kotlin
    import co.amity.socialplus.uikit.AmityUIKitClient

    class MyApplication : Application() {
        override fun onCreate() {
            super.onCreate()
            
            AmityUIKitClient.setup(
                context = this,
                apiKey = "YOUR_API_KEY",
                region = AmityRegion.US // Change to EU or SG
            )
        }
    }
    ```

    **Register a user:**
    ```kotlin
    AmityUIKitClient.registerDevice("user123").subscribe({ 
        // User registered successfully
        Log.d("UIKit", "User registered")
    }, { error ->
        // Handle error
        Log.e("UIKit", "Registration failed", error)
    })
    ```
  </Tab>

  <Tab title="React Native">
    **In your App.js or index.js:**
    
    ```javascript
    import { AmityUIKitProvider } from './amity-react-native-social-ui-kit';

    const App = () => {
      return (
        <AmityUIKitProvider
          apiKey="YOUR_API_KEY"
          apiRegion="us" // Change to "eu" or "sg"
          userId="user123"
          displayName="John Doe"
        >
          {/* Your app content */}
        </AmityUIKitProvider>
      );
    };
    ```
  </Tab>

  <Tab title="Web (React)">
    **Wrap your app with the provider:**
    
    ```jsx
    import { AmityUIKitProvider } from '@amityco/ui-kit-react';
    import '@amityco/ui-kit-react/dist/index.css';

    function App() {
      return (
        <AmityUIKitProvider
          apiKey="YOUR_API_KEY"
          apiRegion="us" // Change to "eu" or "sg"
          userId="user123"
          displayName="John Doe"
        >
          <YourAppContent />
        </AmityUIKitProvider>
      );
    }
    ```
  </Tab>

  <Tab title="Flutter">
    **In your main.dart:**
    
    ```dart
    import 'package:amity_uikit_beta_service/amity_uikit_beta_service.dart';

    void main() async {
      WidgetsFlutterBinding.ensureInitialized();
      
      await AmityUIKit.setup(
        apikey: "YOUR_API_KEY",
        region: AmityRegion.US, // Change to EU or SG
      );
      
      runApp(MyApp());
    }

    // Register user
    await AmityUIKit.registerDevice(userId: "user123");
    ```
  </Tab>
</Tabs>

## Social Feed

<Tabs>
  <Tab title="iOS (Swift)">
    ```swift

    class ViewController: UIViewController {
        override func viewDidLoad() {
            super.viewDidLoad()
            
            // Create and present the social feed
            let feedViewController = AmityFeedViewController()
            
            // Option 1: Present modally
            present(feedViewController, animated: true)
            
            // Option 2: Push to navigation stack
            // navigationController?.pushViewController(feedViewController, animated: true)
            
            // Option 3: Embed in container view
            // addChild(feedViewController)
            // view.addSubview(feedViewController.view)
            // feedViewController.didMove(toParent: self)
        }
    }
    ```

  </Tab>

  <Tab title="Android (Kotlin)">

## Chat System

Implement real-time messaging with chat components:

<Tabs>
  <Tab title="iOS (Swift)">
    ```swift
    import AmityUIKit

    class ChatViewController: UIViewController {
        override func viewDidLoad() {
            super.viewDidLoad()
            
            // Create chat list view
            let chatListVC = AmityChatHomePageViewController()
            
            // Present the chat interface
            present(chatListVC, animated: true)
            
            // Or embed in navigation
            navigationController?.pushViewController(chatListVC, animated: true)
        }
        
        // To open a specific conversation
        func openConversation(withUserId userId: String) {
            let conversationVC = AmityConversationViewController(userId: userId)
            navigationController?.pushViewController(conversationVC, animated: true)
        }
    }
    ```
  </Tab>

  <Tab title="Android (Kotlin)">
    ```kotlin
    import co.amity.socialplus.uikit.chat.home.AmityChatHomePageFragment
    import co.amity.socialplus.uikit.chat.conversation.AmityConversationFragment

    class ChatActivity : AppCompatActivity() {
        override fun onCreate(savedInstanceState: Bundle?) {
            super.onCreate(savedInstanceState)
            setContentView(R.layout.activity_chat)
            
            // Show chat list
            val chatHomeFragment = AmityChatHomePageFragment()
            
            supportFragmentManager.beginTransaction()
                .replace(R.id.fragment_container, chatHomeFragment)
                .commit()
        }
        
        // To open a specific conversation
        fun openConversation(userId: String) {
            val conversationFragment = AmityConversationFragment.newInstance(userId)
            
            supportFragmentManager.beginTransaction()
                .replace(R.id.fragment_container, conversationFragment)
                .addToBackStack(null)
                .commit()
        }
    }
    ```
  </Tab>

  <Tab title="React Native">
    ```jsx
    import { AmityChatUIKit } from './amity-react-native-social-ui-kit';

    const ChatScreen = () => {
      return (
        <AmityChatUIKit
          // Chat interface is automatically rendered
        />
      );
    };

    // To open a specific conversation
    const ConversationScreen = ({ userId }) => {
      return (
        <AmityConversationUIKit
          userId={userId}
        />
      );
    };
    ```
  </Tab>

  <Tab title="Web (React)">
    ```jsx
    import { AmityChatComponent, AmityConversationComponent } from '@amityco/ui-kit-react';

    function ChatPage() {
      const [selectedConversation, setSelectedConversation] = useState(null);
      
      return (
        <div className="chat-page">
          <div className="chat-sidebar">
            <AmityChatComponent 
              onConversationSelect={setSelectedConversation}
            />
          </div>
          
          <div className="chat-main">
            {selectedConversation ? (
              <AmityConversationComponent 
                conversationId={selectedConversation}
              />
            ) : (
              <div className="no-conversation">
                Select a conversation to start chatting
              </div>
            )}
          </div>
        </div>
      );
    }
    ```
  </Tab>

  <Tab title="Flutter">
    ```dart

    }
    ```

## Test Your Implementation

Verify that your component is working correctly:

<Steps>
  <Step title="Build and Run">
    Build your application and run it on a device or emulator.
  </Step>
  
  <Step title="Check Authentication">
    Verify that the user is authenticated and the component loads without errors.
  </Step>
  
  <Step title="Test Functionality">
    Try basic interactions:
    - **Social Feed**: Create a post, like, comment
    - **Chat**: Send messages, view conversations  
    - **Communities**: Browse, join a community
    - **Profiles**: View user information, follow/unfollow
  </Step>
  
  <Step title="Check Console Logs">
    Look for any error messages in your development console or device logs.
  </Step>
</Steps>
