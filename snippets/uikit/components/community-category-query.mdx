List<AmityCommunityCategory> _categories = [];
  late PagingController<AmityCommunityCategory> _controller;
  final scrollController = ScrollController();

  void queryCategories() {
    _controller = PagingController(
      pageFuture: (token) => AmitySocialClient.newCommunityRepository()
          .getCategories()
          .sortBy(AmityCommunityCategorySortOption.NAME)
          .getPagingData(token: token, limit: 20),
      pageSize: 20,
    )..addListener(() {
        if (_controller.error == null) {
          _categories
            ..clear()
            ..addAll(_controller.loadedItems);
        }
      });

    _controller.fetchNextPage();

    scrollController.addListener(() {
      if (scrollController.position.pixels ==
              scrollController.position.maxScrollExtent &&
          _controller.hasMoreItems) {
        _controller.fetchNextPage();
      }
    });
  }
