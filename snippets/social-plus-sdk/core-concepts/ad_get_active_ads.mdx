import { Client } from '@amityco/ts-sdk'

function getActiveAds(placement: string) {
  const disposer = Client.getAdRepository().getActiveAds(
    { placement },
    ({ data: ads, loading, error }) => {
      if (loading) return
      if (error) {
        console.error('Failed to fetch active ads', error)
        return
      }
      console.log('Active ads:', ads)
    },
  )

  return disposer
}
