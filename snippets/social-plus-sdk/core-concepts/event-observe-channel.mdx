import { ChannelRepository } from '@amityco/ts-sdk'

function observeChannelEvents() {
  const unsubscribeCreated = ChannelRepository.onChannelCreated((channel) => {
    console.log('Channel created:', channel.channelId)
  })

  const unsubscribeUpdated = ChannelRepository.onChannelUpdated((channel) => {
    console.log('Channel updated:', channel.channelId)
  })

  const unsubscribeDeleted = ChannelRepository.onChannelDeleted((channel) => {
    console.log('Channel deleted:', channel.channelId)
  })

  return () => {
    unsubscribeCreated()
    unsubscribeUpdated()
    unsubscribeDeleted()
  }
}
