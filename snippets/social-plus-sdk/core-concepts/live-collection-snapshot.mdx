async function liveCollectionSnapshot() {
  const channelId = 'channel-id-123'
  const liveCollection = MessageRepository.getMessages({
    subChannelId: channelId,
    limit: 20,
  })

  let currentSnapshot: Amity.Message[] = []
  const unsubscribe = liveCollection.subscribe(({ data: messages, loading }) => {
    if (!loading) {
      currentSnapshot = messages
      console.log('Snapshot updated, items:', currentSnapshot.length)
    }
  })

  return { unsubscribe, getSnapshot: () => currentSnapshot }
}
