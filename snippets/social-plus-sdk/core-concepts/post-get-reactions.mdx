function getPostReactions(postId: string) {
  const disposer = ReactionRepository.getReactions(
    { referenceId: postId, referenceType: 'post' },
    ({ data: reactions, loading, error }) => {
      if (loading) return
      if (error) {
        console.error('Failed to fetch reactions', error)
        return
      }
      console.log('Post reactions:', reactions)
    },
  )

}
