import { FileRepository } from '@amityco/ts-sdk'

async function getVideoThumbnail() {
  const fileId = 'video-file-id-123'
  const liveObject = FileRepository.getFile(fileId)
  const unsubscribe = liveObject.subscribe(({ data: file, loading }) => {
    if (!loading && file) {
      const thumbnailUrl = file.attributes?.thumbnails?.['720p'] ?? file.fileUrl
      console.log('Video thumbnail URL:', thumbnailUrl)
    }
  })
  return unsubscribe
}
