fun unsyncUserPresence(
        userPresenceRepository: AmityUserPresenceRepository,
        userId: String
    ) {
        userPresenceRepository.unsyncUserPresence(
            userId = userId,
            viewId = "" // optional, can use hash code of the View Holder class as view id
            //  recommended to use view id due to the Recycler View item update behavior
        )
    }

    //  Example usage about unsyncing user presence inside RecyclerView Adapter
    fun onViewDetachedToWindow(holder: UserViewHolder) {
        //  when View Holder is detached from adapter, stop syncing presence for detached channel
        holder.unsyncUserPresence()
    }

    class UserViewHolder(view: View) : RecyclerView.ViewHolder(view) {

        private var userId: String = ""

        fun bind(user: AmityUser) {
            userId = user.getUserId()
        }

        fun unsyncUserPresence() {
            AmityCoreClient.newUserPresenceRepository()
                .unsyncUserPresence(
                    userId = userId,
                    viewId = hashCode().toString()
                )
        }
    }
