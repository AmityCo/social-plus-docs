import { PostRepository } from '@amityco/ts-sdk'

function observePost(postId: string) {
  const unsubscribe = PostRepository.getPost(postId, ({ data: post, loading, error }) => {
    if (loading) return
    if (error) {
      console.error('Error fetching post:', error)
      return
    }

    console.log('Post data:', post.data)
    console.log('Reactions count:', post.reactionsCount)
    console.log('Comments count:', post.commentsCount)
  })

  return unsubscribe
}
