func adsQuery() {
        let adRepository = AmityAdRepository(client: AmityManager.shared.client!)
        let queryOptions = AmityAdQueryOptions()
        queryOptions.placement = .feed
        
        let liveCollection = adRepository.getAds(options: queryOptions)
        let token = liveCollection.observe { collection, changes, error in
            guard error == nil else { return }
            print("Ads count: \(collection.count())")
            for ad in collection.allObjects() {
                print("Ad: \(ad.adId), type: \(ad.adType)")
            }
        }
        // Retain token as long as observation is needed
    }

void adsQuery() {
    // Query ads using a paging controller
    final controller = PagingController<AmityAd>(
      pageFuture: (token) => AmityAdRepository()
          .queryAds()
          .getPagingData(token: token, limit: 20),
      pageSize: 20,
    );
    controller.addListener(() {
      final ads = controller.itemList ?? [];
      print('Ads loaded: \${ads.length}');
    });
    controller.fetchNextPage();
  }
