import { MessageRepository } from '@amityco/ts-sdk'

async function liveCollectionWithRecyclerView() {
  const channelId = 'channel-id-123'
  const liveCollection = MessageRepository.getMessages({
    subChannelId: channelId,
    limit: 20,
  })

  // Bind to a virtual list (RecyclerView equivalent in web)
  const listEl = document.getElementById('message-list')!
  const unsubscribe = liveCollection.subscribe(({ data: messages, loading }) => {
    if (!loading) {
      listEl.innerHTML = messages
        .map(m => `<div class="message-item">${m.data?.text ?? ''}</div>`)
        .join('')
    }
  })

  return unsubscribe
}
