        var token: AmityNotificationToken?
        let collection = messageRepository.getMessages(
            channelId: "channelId",
            subChannelId: nil,
            filterByParentId: false,
            parentId: nil,
            orderBy: .descending,
            includeDeleted: false
        )
        token = collection.observe { liveCollection, _, error in
            if let error = error {
                print("Error loading messages: \(error.localizedDescription)")
                return
            }
            let messages = (0..<liveCollection.count()).compactMap { liveCollection.object(at: $0) }
            let mentionedMessages = messages.filter { msg in
                guard let metadata = msg.metadata,
                      let mentions = metadata["mentioned"] as? [[String: Any]] else { return false }
                return !mentions.isEmpty
            }
            print("Messages with mentions: \(mentionedMessages.count)")
        }
