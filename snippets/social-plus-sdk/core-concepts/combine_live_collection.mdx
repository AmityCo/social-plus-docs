void combineLiveCollection() {
    // Combine multiple live collections into a single stream
    final postCollection = AmitySocialClient.newFeedRepository()
        .getGlobalFeed()
        .getLiveCollection(pageSize: 20);

    final communityCollection = AmitySocialClient.newCommunityRepository()
        .getCommunities()
        .getLiveCollection(pageSize: 20);

    Rx.combineLatest2(
      postCollection.asStream(),
      communityCollection.asStream(),
      (posts, communities) => (posts: posts, communities: communities),
    ).listen((combined) {
      print('Posts: \${combined.posts.length}, Communities: \${combined.communities.length}');
    });
  }
