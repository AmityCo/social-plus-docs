import { UserRepository } from '@amityco/ts-sdk'

function queryUsersWithKeyword(keyword: string) {
  const disposer = UserRepository.searchUserByDisplayName(
    { displayName: keyword, limit: 20 },
    ({ data: users, loading, error }) => {
      if (loading) return
      if (error) {
        console.error('Failed to search users', error)
        return
      }
      console.log('Users found:', users)
    },
  )

  return disposer
}
