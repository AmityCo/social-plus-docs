func adsCreate() {
        let adRepository = AmityAdRepository(client: AmityManager.shared.client!)
        let builder = AmityAdCreationBuilder()
        builder.setAdType(.banner)
        builder.setTargetAudience(.all)
        builder.setPlacement(.feed)
        
        adRepository.createAd(builder: builder) { ad, error in
            if let error = error {
                print("Ad creation failed: \(error)")
                return
            }
            print("Ad created, adId: \(ad?.adId ?? "")")
        }
    }
