import { CommentRepository } from '@amityco/ts-sdk'

async function createCommentWithMentionApi(referenceId: string, text: string, userId: string) {
  try {
    const { data: comment } = await CommentRepository.createComment({
      referenceId,
      referenceType: 'post',
      data: { text: `Hello @${userId} ${text}` },
      mentionees: [{ type: 'user', userIds: [userId] }],
      metadata: { hasUserMention: true },
    });
    console.log('Comment with mention API created:', comment.commentId);
