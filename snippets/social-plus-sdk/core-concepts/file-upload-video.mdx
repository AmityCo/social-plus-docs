                when (result) {

                        val progress = result.getUploadInfo().getProgressPercentage()

                        val video: AmityVideo = result.getFile()
                    }
                    is AmityUploadResult.ERROR -> {
                        val error = AmityError.from(result.getError())
                    }
                    is AmityUploadResult.CANCELLED -> { }
                }
            }
            .subscribe()
    }
