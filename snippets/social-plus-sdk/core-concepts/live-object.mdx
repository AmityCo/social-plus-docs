import { PostRepository } from '@amityco/ts-sdk'

function observeLiveObject(postId: string) {
  const unsubscribe = PostRepository.getPost(
    postId,
    ({ data: post, loading, error }) => {
      if (loading) return
      if (error) {
        console.error('Error observing post:', error)
        return
      }
      console.log('Post live object updated:', post.postId)
      console.log('Post text:', post.data?.text)
    },
  )

  // Call unsubscribe() when done to stop receiving updates
  return unsubscribe
}
