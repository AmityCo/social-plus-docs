import { analytics } from '@amityco/ts-sdk'

function getViewedUsersForPost(postId: string) {
  const unsubscribe = analytics.getViewedUsers(
    { referenceId: postId, referenceType: 'post' },
    ({ data: users, loading, error }) => {
      if (loading) return
      if (error) {
        console.error('Error fetching viewed users:', error)
        return
      }
      console.log(`Users who viewed post ${postId}:`, users.map((u) => u.userId))
    },
  )

  return unsubscribe
}
