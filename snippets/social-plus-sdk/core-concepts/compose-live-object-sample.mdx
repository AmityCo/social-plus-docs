import { PostRepository } from '@amityco/ts-sdk'

function composeLiveObjectSample(postId: string) {
  const unsubscribe = PostRepository.getPost(postId, ({ data: post, loading, error }) => {
    if (error) {
      console.error('Live object error:', error)
      return
    }
    if (!loading && post) {
      console.log('Live object post updated:', post.postId, post.dataType)
    }
  })

  // Stop listening when no longer needed
  return unsubscribe
}
