import { PostRepository } from '@amityco/ts-sdk'

function observePostCollection() {
  const liveCollection = PostRepository.getPosts({ targetType: 'community', targetId: 'communityId' })

  liveCollection.on('dataUpdated', (posts) => {
    console.log('Posts updated:', posts.length)
  })

  liveCollection.on('dataError', (error) => {
    console.error('Error fetching posts:', error)
  })

  return () => liveCollection.dispose()
}
