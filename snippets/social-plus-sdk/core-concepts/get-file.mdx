        let fileRepository = AmityFileRepository(client: AmityManager.shared.client!)
        let url = URL(string: "https://example.com/sample.pdf")!
        let fileData = try! Data(contentsOf: url)
        fileRepository.uploadFile(fileData, fileExtension: "pdf", progress: { progress in
            print("Upload progress: \(progress)")
        }) { file, error in
            if let error = error { print("Error: \(error)"); return }
            print("File uploaded: \(file?.fileId ?? "")")
