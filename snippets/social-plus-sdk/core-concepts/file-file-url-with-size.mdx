fun getFileUrlWithSize(fileId: String) {
        AmityCoreClient.newFileRepository()
            .getFile(fileId = fileId)
            .doOnSuccess { rawFile: AmityRawFile ->
                if (rawFile.getFileType() == AmityFileType.IMAGE) {
                    val image: AmityImage? = rawFile.asAmityImage()
                    image?.let {
                        val size = AmityImage.Size.MEDIUM // SMALL, MEDIUM, LARGE, FULL
                        val imageUrl = it.getUrl(size)
                    }
                }
            }
            .doOnError {
                // Exception
            }
            .subscribe()
    }
