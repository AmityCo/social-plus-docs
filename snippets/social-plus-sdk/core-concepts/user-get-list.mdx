import { UserRepository } from '@amityco/ts-sdk'

function getUserList() {
  const users = []

  const unsubscribe = UserRepository.getUsers(
    { displayName: '' },
    (response) => {
      if (response.error) throw response.error
      if (!response.loading) {
        users.length = 0
        users.push(...response.data)
        console.log('Users:', users)
      }
    },
  )

  return unsubscribe
}
