import { CommunityRepository } from '@amityco/ts-sdk'

function observeLiveCollection() {
  const communities = []

  const unsubscribe = CommunityRepository.getCommunities(
    { limit: 20 },
    (response) => {
      if (response.error) {
        console.error('Error:', response.error)
        return
      }
      if (!response.loading) {
        communities.length = 0
        communities.push(...response.data)
        console.log('Live collection updated:', communities.length, 'items')
      }
    },
  )

  // Call unsubscribe() when done to stop receiving updates
  return unsubscribe
}
