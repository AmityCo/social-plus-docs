```swift iOS
import UIKit
import AmitySDK

@UIApplicationMain
class ExampleAppDelegate: UIResponder, UIApplicationDelegate {

    var client: AmityClient!
    
    func application(_ application: UIApplication, didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?) -> Bool {
        // Setup AmityClient
        do {
            client = try AmityClient(apiKey: "<api-key>")
        } catch {
            fatalError("Unable to initialize client properly: \(error.localizedDescription)")
        }
        return true
    }
    
    // MARK: - Push Notification
    
    func application(_ application: UIApplication, didRegisterForRemoteNotificationsWithDeviceToken deviceToken: Data) {
        
        // Prepare device token
        // Source: https://developer.apple.com/documentation/usernotifications/registering_your_app_with_apns
    
        // Transform deviceToken into a raw string, before sending to sdk server.
        let tokenParts = deviceToken.map { data in String(format: "%02.2hhx", data) }
        let tokenString = tokenParts.joined()
        
        // Send device token to sdk server
        client.registerDeviceForPushNotification(withDeviceToken: tokenString)
    }
    
    func application(_ application: UIApplication, didFailToRegisterForRemoteNotificationsWithError error: Error) {
        // Handler error
        print("Unable to register remote notification: \(error)")
    }
    
}
```
