import { PostRepository } from '@amityco/ts-sdk'

function observePostEvents() {
  const unsubscribeCreated = PostRepository.onPostCreated((post) => {
    console.log('Post created:', post.postId)
  })

  const unsubscribeUpdated = PostRepository.onPostUpdated((post) => {
    console.log('Post updated:', post.postId)
  })

  const unsubscribeDeleted = PostRepository.onPostDeleted((post) => {
    console.log('Post deleted:', post.postId)
  })

  return () => {
    unsubscribeCreated()
    unsubscribeUpdated()
    unsubscribeDeleted()
  }
}
