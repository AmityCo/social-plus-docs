import { ReactionRepository } from '@amityco/ts-sdk'

function getCommentReactions(commentId: string) {
  const liveCollection = ReactionRepository.queryReactions({
    referenceType: 'comment',
    referenceId: commentId,
  })

  liveCollection.on('dataUpdated', (reactions) => {
    console.log(`Reactions on comment ${commentId}:`, reactions.length)
    reactions.forEach((reaction) => {
      console.log(' -', reaction.reactionName, 'by', reaction.userId)
    })
  })

  liveCollection.on('dataError', (error) => {
    console.error('Error fetching comment reactions:', error)
  })
}
