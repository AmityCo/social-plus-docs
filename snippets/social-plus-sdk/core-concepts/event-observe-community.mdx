import { CommunityRepository } from '@amityco/ts-sdk'

function observeCommunityEvents() {
  const unsubscribeCreated = CommunityRepository.onCommunityCreated((community) => {
    console.log('Community created:', community.communityId)
  })

  const unsubscribeUpdated = CommunityRepository.onCommunityUpdated((community) => {
    console.log('Community updated:', community.communityId)
  })

  const unsubscribeDeleted = CommunityRepository.onCommunityDeleted((community) => {
    console.log('Community deleted:', community.communityId)
  })

  return () => {
    unsubscribeCreated()
    unsubscribeUpdated()
    unsubscribeDeleted()
  }
}
