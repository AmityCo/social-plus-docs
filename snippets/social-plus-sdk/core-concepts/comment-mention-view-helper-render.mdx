fun commentMentionViewHelperRender() {
        val mentionHelper = com.amityco.socialcloud.sdk.helper.core.mention.AmityMentionMetadataGetter(
            org.json.JSONObject(comment.getMetadata().toString())
        )
        val mentionedUsers = mentionHelper.getMentionedUsers()
        val spannableString = android.text.SpannableString(comment.getData().toString())
        mentionedUsers.forEach { mention ->
            val start = mention.index
            val end = start + mention.length + 1
            spannableString.setSpan(
                android.text.style.ForegroundColorSpan(android.graphics.Color.BLUE),
                start, end, android.text.Spannable.SPAN_EXCLUSIVE_EXCLUSIVE
            )
        }
    }
