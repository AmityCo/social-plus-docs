fun downloadImageAsData(fileId: String) {
        AmityCoreClient.newFileRepository()
            .getFile(fileId = fileId)
            .doOnSuccess { rawFile: AmityRawFile ->
                if (rawFile.getFileType() == AmityFileType.IMAGE) {
                    val image: AmityImage? = rawFile.asAmityImage()
                    image?.let {
                        val smallUrl = it.getUrl(AmityImage.Size.SMALL)
                        val mediumUrl = it.getUrl(AmityImage.Size.MEDIUM)
                        val largeUrl = it.getUrl(AmityImage.Size.LARGE)
                        val fullUrl = it.getUrl(AmityImage.Size.FULL)
                    }
                }
            }
            .doOnError {
                // Exception
            }
            .subscribe()
    }
