function composeLiveCollectionSample(communityId: string) {
  const unsubscribe = PostRepository.getPosts(
    { targetId: communityId, targetType: 'community', sortBy: 'lastCreated' },
    ({ data: posts, loading, hasNextPage, onNextPage, error }) => {
      if (error) {
        console.error('Live collection error:', error)
        return
      }
      if (!loading) {
        console.log('Live collection posts:', posts.length)
        posts.forEach((post) => console.log(post.postId, post.dataType))
        if (hasNextPage) onNextPage()
      }
    },
  )

  // Stop listening when no longer needed
  return unsubscribe
}
