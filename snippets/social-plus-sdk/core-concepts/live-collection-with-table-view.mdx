import { CommunityRepository } from '@amityco/ts-sdk'

async function liveCollectionWithTableView() {
  const communityId = 'community-id-123'
  const liveCollection = CommunityRepository.getMembers({
    communityId,
    limit: 20,
  })

  // Bind to a table view (equivalent rendering in web)
  const tableBody = document.getElementById('member-table-body')!
  const unsubscribe = liveCollection.subscribe(({ data: members, loading }) => {
    if (!loading) {
      tableBody.innerHTML = members
        .map(m => `<tr><td>${m.userId}</td><td>${m.displayName}</td></tr>`)
        .join('')
    }
  })

  return unsubscribe
}
