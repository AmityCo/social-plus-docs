// Parse mention metadata to highlight mentions in UI
        let text = "Hello @john and @jane!"
        let metadata: [String: Any] = [
            "mentioned": [
                ["type": "user", "userId": "johnUserId", "length": 5, "index": 6],
                ["type": "user", "userId": "janeUserId", "length": 5, "index": 16]
            ]
        ]
        if let mentionedArray = metadata["mentioned"] as? [[String: Any]] {
            let attributedString = NSMutableAttributedString(string: text)
            for mention in mentionedArray {
                if let index = mention["index"] as? Int,
                   let length = mention["length"] as? Int {
                    let range = NSRange(location: index, length: length)
                    attributedString.addAttribute(.foregroundColor,
                        value: UIColor.systemBlue, range: range)
                }
            }
            print("Highlighted text: \(attributedString.string)")
        }
