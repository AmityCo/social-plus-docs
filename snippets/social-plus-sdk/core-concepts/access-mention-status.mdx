function checkMentionStatus(channelId: string) {
  const unsubscribe = ChannelRepository.getChannel(channelId, ({ data: channel, loading }) => {
    if (loading) return

    const { isMentioned } = channel
    if (isMentioned) {
      console.log('You have been mentioned in this channel')
    } else {
      console.log('No unread mentions in this channel')
