fun uploadClip(clipUri: Uri) {
        val uploadId = UUID.randomUUID().toString()
        AmityCoreClient.newFileRepository()
            .uploadClip(
                uploadId = uploadId,
                uri = clipUri,
                contentFeedType = AmityContentFeedType.POST
            )
            .doOnNext { result: AmityUploadResult<AmityClip> ->
                when (result) {
                    is AmityUploadResult.PROGRESS -> {
                        val progress = result.getUploadInfo().getProgressPercentage()
                    }
                    is AmityUploadResult.COMPLETE -> {
                        val clip: AmityClip = result.getFile()
                    }
                    is AmityUploadResult.ERROR -> { /* handle error */ }
                    is AmityUploadResult.CANCELLED -> { }
                }
            }
            .subscribe()
    }
