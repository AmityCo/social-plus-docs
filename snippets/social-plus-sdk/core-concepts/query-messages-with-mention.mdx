import { MessageRepository } from '@amityco/ts-sdk'

async function queryMessagesWithMention() {
  const params = {
    subChannelId: 'subChannelId123',
    sortBy: 'segmentDesc' as const,
    limit: 20,
    tags: ['mention'],
  };

  const liveCollection = MessageRepository.getMessages(params);
  liveCollection.on('dataUpdated', messages => {
    const mentionMessages = messages.filter(m => m.mentionees && m.mentionees.length > 0);
    console.log('Messages with mentions:', mentionMessages.map(m => m.messageId));
  });
}
