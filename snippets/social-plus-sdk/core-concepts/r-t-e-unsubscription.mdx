val subscriptions: MutableList<AmityTopicSubscription> = mutableListOf()
    override fun onViewCreated(view: View, savedInstanceState: Bundle?) {
        super.onViewCreated(view, savedInstanceState)
        observeCommunity() // observe via LiveObect
        subscribeCommunity(community = community) // subscribe to a community
    }
    //
    private fun subscribeCommunity(community: AmityCommunity) {
        val topicSubscription = community.subscription(events = AmityCommunityEvents.COMMUNITY)
        topicSubscription
            .subscribeTopic()
            .doOnComplete {
                // keep ref of topicSubscription for disconnection
                subscriptions.add(topicSubscription)
            }
            .subscribe()
    }
    //
    override fun onDestroy() {
        // Unsubscribe topics before leaving each screen
        subscriptions.forEach { topicSubscription ->
            topicSubscription.unsubscribeTopic()
                .subscribe()
        }
        super.onDestroy()
    }
