    // Combine two live objects and react when either changes
    final livePost = AmitySocialClient.newPostRepository().getPost('postId');
    final liveUser = AmityCoreClient.newUserRepository().getUser('userId');

      liveUser.stream,
      (post, user) => (post: post, user: user),
