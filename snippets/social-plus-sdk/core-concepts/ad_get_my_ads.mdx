import { Client } from '@amityco/ts-sdk'

function getMyAds() {
  const disposer = Client.getAdRepository().getMyAds(
    { limit: 20 },
    ({ data: ads, loading, error }) => {
      if (loading) return
      if (error) {
        console.error('Failed to fetch my ads', error)
        return
      }
      console.log('My ads:', ads)
    },
  )

  return disposer
}
