import { AdRepository } from '@amityco/ts-sdk'

async function markAdLinkAsClicked(adId: string) {
  const networkAds = await AdRepository.getNetworkAds()
  const ad = networkAds.ads.find((a) => a.adId === adId)
  if (ad) {
    ad.analytics.markLinkAsClicked('feed')
    console.log('Ad link marked as clicked:', adId)
  }
}
