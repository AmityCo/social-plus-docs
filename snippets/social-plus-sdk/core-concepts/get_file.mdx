func getFile() {
        let fileRepository = AmityFileRepository(client: AmityManager.shared.client!)
        
        fileRepository.getFile(withFileId: "fileId") { file, error in
            if let error = error {
                print("Failed to get file: \(error)")
                return
            }
            guard let file = file else { return }
            print("File: \(file.fileId), name: \(file.attributes["name"] as? String ?? ""), size: \(file.fileSize)")
        }
    }
