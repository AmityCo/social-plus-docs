async function postMentionViewHelperRender() {
  const { PostRepository } = Amity;
  const { data: post } = await PostRepository.getPost('postId');
  const metadata = post.metadata ?? {};
  const mentionedUsers = metadata.mentioned ?? [];
  const textContent = (post.data as Amity.TextData).text ?? '';
  const rendered = mentionedUsers.reduce((acc: string, m: any) => {
    return acc.replace(textContent.slice(m.index, m.index + m.length), `@${m.userId}`);
  }, textContent);
  console.log('Helper rendered text:', rendered);
}
