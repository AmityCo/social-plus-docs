fun getLiveReactions(
        postId: String,
    ) {
        AmityCoreClient.newLiveReactionRepository()
            .getReactions(
                referenceId = postId,
                referenceType = AmityLiveReactionReferenceType.POST,
            )
            .subscribeOn(Schedulers.io())
            .observeOn(AndroidSchedulers.mainThread())
            .doOnNext { reactions: List<AmityLiveReaction> ->
                // Handle the list of reactions
                reactions.forEach { reaction: AmityLiveReaction ->
                    reaction.getReactionName() // Reaction name i.g. "like"
                }
            }
            .subscribe()
}
