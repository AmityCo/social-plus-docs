import { MessageRepository } from '@amityco/ts-sdk'

function getLiveStreamChat(channelId: string) {
  const liveCollection = MessageRepository.getMessages({ channelId })

  liveCollection.on('dataUpdated', (messages) => {
    console.log('Live stream chat messages:', messages.length)
    messages.forEach((message) => {
      console.log('Message:', message.data?.text)
    })
  })

  liveCollection.on('dataError', (error) => {
    console.error('Error fetching live stream chat:', error)
  })

  return () => liveCollection.dispose()
}

// Example usage
getLiveStreamChat('channelId')
