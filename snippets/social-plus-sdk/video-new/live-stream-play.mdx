import { StreamRepository } from '@amityco/ts-sdk'

function playLiveStream(streamId: string) {
  const unsubscribe = StreamRepository.getStream(streamId, ({ data: stream, loading, error }) => {
    if (loading) return
    if (error) {
      console.error('Error fetching stream:', error)
      return
    }

    if (stream.status === 'live') {
      console.log('Stream is live, play URL:', stream.watcherUrl)
    } else {
      console.log('Stream is not live, status:', stream.status)
    }
  })

  return unsubscribe
}

// Example usage
playLiveStream('streamId')
