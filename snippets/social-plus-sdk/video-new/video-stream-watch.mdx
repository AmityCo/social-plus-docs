function watchVideoStream(streamId: string) {
  const disposer = StreamRepository.getStream(streamId, ({ data: stream, loading, error }) => {
    if (loading) return
    if (error) {
      console.error('Failed to fetch stream', error)
      return
    }
    if (stream?.watcherUrls?.length) {
      console.log('Watch URL:', stream.watcherUrls[0])
    }
  })

}
