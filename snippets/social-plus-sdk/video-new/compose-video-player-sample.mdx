import { StreamRepository } from '@amityco/ts-sdk'

function composeVideoPlayerSample(streamId: string) {
  const unsubscribe = StreamRepository.getStreamById(
    streamId,
    ({ data: stream, loading, error }) => {
      if (error) {
        console.error('Error fetching stream:', error)
        return
      }
      if (!loading && stream) {
        console.log('Stream status:', stream.status)
        console.log('Viewer URL (HLS):', stream.watcherUrl)
      }
    },
  )

  return unsubscribe
}
