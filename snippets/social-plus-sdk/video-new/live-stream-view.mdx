import { StreamRepository } from '@amityco/ts-sdk'

async function liveStreamView(streamId: string) {
  const liveObject = StreamRepository.getStreamById(streamId)
  liveObject.on('dataUpdated', (stream) => {
    if (stream.status === 'live') {
      console.log('Watch URL:', stream.watcherUrl)
    }
  })
  return liveObject
}
