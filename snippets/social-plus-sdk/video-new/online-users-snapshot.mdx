    fun loadMoreUsers(snapshot: AmityOnlineUsersSnapshot) {
        var users: List<AmityUser>

        //  check if snapshot can load more users
        if (snapshot.canLoadMore()) {
            snapshot.loadMore()
                .doOnComplete {
                    //  another 20 users will be loaded and there will be total of 40 users

                }

                }
