import { PostRepository } from '@amityco/ts-sdk'

async function createLivestreamPost(targetId: string, streamId: string, title: string) {
  try {
    const { data: post } = await PostRepository.createPost({
      targetType: 'community',
      targetId,
      dataType: 'liveStream',
      data: { streamId },
      text: title,
      tags: ['livestream'],
    });
    console.log('Livestream post created:', post.postId, 'streamId:', streamId);
    return post;
  } catch (error) {
    console.error('Failed to create livestream post:', error);
  }
}
