function getSyncingChannelPresences() {
  // querySyncingPresenceChannels returns channels with active presence sync
  const liveCollection = ChannelRepository.querySyncingPresenceChannels()

  liveCollection.on('dataUpdated', (channels) => {
    console.log(`Currently syncing presence for ${channels.length} channel(s):`)

      console.log(' -', channel.channelId, '| active members:', channel.activeMemberCount)

  })
}
