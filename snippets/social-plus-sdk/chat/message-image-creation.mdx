import { MessageRepository } from '@amityco/ts-sdk'

async function messageImageCreation(channelId: string, imageFileId: string, caption?: string) {
  const message = await MessageRepository.createMessage({
    channelId,
    dataType: 'image',
    data: {
      fileId: imageFileId,
      caption,
    },
  })
  console.log('Image message created:', message.messageId)
  return message
