import { MessageRepository } from '@amityco/ts-sdk'

async function sendCustomMessage(subChannelId: string) {
  const message = await MessageRepository.createMessage({
    subChannelId,
    dataType: 'custom',
    data: {
      type: 'card',
      title: 'Custom Card',
      body: 'This is a custom message payload',
    },
    metadata: { version: 1 },
  })

  console.log('Custom message sent:', message.messageId)
  return message
}
