fun muteChannelMembers(
        channelRepository: AmityChannelRepository,
        channel: AmityChannel
    ) {
        // mute channel members
        val timeout: Duration = Duration.standardMinutes(10)
        val userIds = listOf("user1", "user2")

        channelRepository.moderation(channelId = channel.getChannelId())
            .muteMembers(
                timeout = timeout,
                userIds = userIds
            )
            .subscribe()

        // unmute channel members
        channelRepository.moderation(channelId = channel.getChannelId())
            .muteMembers(
                timeout = Duration.ZERO,
                userIds = userIds
            )
            .subscribe()
    }
