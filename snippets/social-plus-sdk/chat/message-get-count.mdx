import { ChannelRepository } from '@amityco/ts-sdk'

function getMessageCount(channelId: string) {
  const unsubscribe = ChannelRepository.getChannel(
    channelId,
    ({ data: channel, loading, error }) => {
      if (loading) return
      if (error) throw error
      console.log('Unread message count:', channel.unreadCount)
      console.log('Total member count:', channel.memberCount)
    },
  )
  return unsubscribe
}
