import { ChannelRepository } from '@amityco/ts-sdk'

async function banUnbanChannelMember(channelId: string, userId: string, action: 'ban' | 'unban') {
  try {
    if (action === 'ban') {
      await ChannelRepository.Membership.banMembers(channelId, [userId]);
      console.log('User banned from channel:', userId);
    } else {
      await ChannelRepository.Membership.unbanMembers(channelId, [userId]);
      console.log('User unbanned from channel:', userId);
    }
  } catch (error) {
    console.error('Failed to ban/unban member:', error);
  }
}
