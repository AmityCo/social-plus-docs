import { ChannelRepository } from '@amityco/ts-sdk'

async function searchChannels() {
  const params = {
    displayName: 'general',
    filter: 'all' as const,
    sortBy: 'displayName' as const,
  };

  const liveCollection = ChannelRepository.getChannels(params);
  liveCollection.on('dataUpdated', channels => {
    console.log('Search results:', channels.map(c => c.displayName));
  });
  liveCollection.on('dataError', err => console.error('Error searching channels:', err));
}
