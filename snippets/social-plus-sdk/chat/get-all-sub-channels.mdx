import { SubChannelRepository } from '@amityco/ts-sdk'

function getAllSubChannels() {
  const liveCollection = SubChannelRepository.getSubChannels({ channelId: 'channelId' })

  liveCollection.on('dataUpdated', (subChannels) => {
    console.log('Sub-channels:', subChannels)
  })

  liveCollection.on('dataError', (error) => {
    console.error('Failed to get sub-channels:', error)
  })

  return () => liveCollection.dispose()
}
