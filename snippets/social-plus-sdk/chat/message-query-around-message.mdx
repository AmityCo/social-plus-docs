fun jumpToTarget(
        messageRepository: AmityMessageRepository,
        subChannelId: String,
        targetId: String,
        newAdapter: PagingDataAdapter<AmityMessage, *>,
        previousAdapter: PagingDataAdapter<AmityMessage, *>?,
        recyclerView: RecyclerView,
        lifecycle: Lifecycle
    ) {
        /*
            The application logic for jumping to target message
            performs swapping of adapters depending on LoadStates

            The logic can be either
            1. Assign new adapter to recyclerView, and fallback to previous adapter upon error conditions
            or
            2. Only assign new adapter to recyclerView upon success query

            The sample code below is for the 1st logic
        */

        // Error handling by observing adapter's load state
        var listener: ((CombinedLoadStates) -> Unit)? = null
        listener = {
            val refreshState = it.refresh
            if (refreshState is LoadState.Error) {
                newAdapter.removeLoadStateListener(listener!!)

                val error = AmityError.from(refreshState.error)
                val shouldFallback =
                    (error != AmityError.CONNECTION_ERROR) && previousAdapter != null
                val shouldRefresh =
                    (error != AmityError.CONNECTION_ERROR) && previousAdapter == null

                if (shouldFallback) {
                    recyclerView.adapter = previousAdapter
                } else if (shouldRefresh) {
                    // recover by refreshing, query without aroundMessageId instead
                } else {
                    // when no network connection, available cache will be submitted to the adapter
                }
            }
        }

        // Perform query with the new adapter
        recyclerView.adapter = newAdapter
        newAdapter.addLoadStateListener(listener)
        messageRepository.getMessages(subChannelId = subChannelId)
            .aroundMessageId(messageId = targetId)
            .build()
            .query()
            .doOnNext {
                newAdapter.submitData(lifecycle, it)
            }
            .doOnError {
                // Exception
            }
            .subscribe()
    }
