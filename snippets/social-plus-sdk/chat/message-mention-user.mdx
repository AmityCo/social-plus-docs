import { MessageRepository } from '@amityco/ts-sdk'

async function messageMentionUser(channelId: string, mentionedUserId: string) {
  const message = await MessageRepository.createMessage({
    channelId,
    dataType: 'text',
    data: { text: `Hey @${mentionedUserId}, how are you?` },
    mentionees: [{ type: 'user', userIds: [mentionedUserId] }],
    metadata: {},
  })
  console.log('User mention message sent:', message.messageId)
  return message
}
