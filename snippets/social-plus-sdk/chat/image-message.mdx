import { FileRepository, MessageRepository } from '@amityco/ts-sdk'

async function imageMessage() {
  const channelId = 'channel-id-123'
  const fileInput = document.getElementById('fileInput') as HTMLInputElement
  const file = fileInput.files?.[0]
  if (!file) return

  const uploadResult = await FileRepository.uploadImage(file)
  const { data: message, error } = await MessageRepository.createMessage({
    subChannelId: channelId,
    dataType: 'image',
    data: { fileId: uploadResult.data?.fileId ?? '' },
  })
  if (error) {
    console.error('Failed to send image message:', error)
  } else {
    console.log('Image message sent:', message)
  }
}
