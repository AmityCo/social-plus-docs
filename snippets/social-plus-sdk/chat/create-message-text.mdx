import { MessageRepository } from '@amityco/ts-sdk'

async function createMessageText(channelId: string, text: string) {
  try {
    const { data: message } = await MessageRepository.createMessage({
      channelId,
      dataType: 'text',
      data: { text },
    });
    console.log('Text message sent:', message.messageId, 'syncState:', message.syncState);
    return message;
  } catch (error) {
    console.error('Failed to create text message:', error);
  }
}
