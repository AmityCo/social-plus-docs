import { ChannelRepository } from '@amityco/ts-sdk'

function checkChannelPermission(channelId: string, requiredRole: string) {
  const unsubscribe = ChannelRepository.getChannel(channelId, ({ data: channel, loading }) => {
    if (loading) return

    const currentUserRoles = channel.currentUserMembership?.roles ?? []
    const hasPermission = currentUserRoles.includes(requiredRole)

    if (hasPermission) {
      console.log(`User has role '${requiredRole}' in channel ${channelId}`)
    } else {
      console.log(`User does not have role '${requiredRole}' in channel ${channelId}`)
    }
  })

  return unsubscribe
}

// Example usage
checkChannelPermission('channelId', 'channel-moderator')
