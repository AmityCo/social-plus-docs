import { ChannelRepository } from '@amityco/ts-sdk'

async function createCommunityChannel(communityId: string, channelName: string) {
  try {
    const { data: channel } = await ChannelRepository.createChannel({
      displayName: channelName,
      type: 'community',
      tags: [communityId],
      metadata: { linkedCommunityId: communityId },
    });
    console.log('Community channel created:', channel.channelId);
    return channel;
  } catch (error) {
    console.error('Failed to create community channel:', error);
  }
}
