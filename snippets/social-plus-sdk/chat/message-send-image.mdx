import { MessageRepository } from '@amityco/ts-sdk'

async function messageSendImage(channelId: string, imageFileId: string, caption?: string) {
  const message = await MessageRepository.createMessage({
    channelId,
    dataType: 'image',
    data: {
      fileId: imageFileId,
      caption,
    },
  })
  console.log('Image message sent:', message.messageId)
  return message
}
