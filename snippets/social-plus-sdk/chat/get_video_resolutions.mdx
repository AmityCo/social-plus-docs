import { MessageRepository, FileRepository } from '@amityco/ts-sdk'

async function getVideoResolutions() {
  const messageId = 'video-message-id-123'
  const liveObject = MessageRepository.getMessage(messageId)
  const unsubscribe = liveObject.subscribe(({ data: message, loading }) => {
    if (!loading && message && message.dataType === 'video') {
      const videoData = message.data as { fileId: string }
      const fileObject = FileRepository.getFile(videoData.fileId)
      fileObject.subscribe(({ data: file }) => {
        console.log('Available resolutions:', file?.videoAttributes?.quality)
      })
    }
  })
  return unsubscribe
}
