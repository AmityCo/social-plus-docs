import { MessageRepository } from '@amityco/ts-sdk'

async function sendAudioMessage(subChannelId: string, audioFileId: string) {
  const message = await MessageRepository.createMessage({
    subChannelId,
    dataType: 'audio',
    data: { fileId: audioFileId },
  })

  console.log('Audio message sent:', message.messageId)
  return message
}
