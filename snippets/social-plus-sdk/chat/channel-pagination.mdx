func channelPagination() {
        let client = AmityClient.shared
        let channelRepository = AmityChannelRepository(client: client)
        let query = AmityChannelQuery()
        query.filter = .member
        let liveCollection = channelRepository.getChannels(query)
        liveCollection.observe { collection, _, error in
            guard error == nil else { return }
            let channels = (0..<collection.count()).compactMap { collection.object(at: $0) }
            print("Loaded \(channels.count) channels")
            if collection.hasNext {
                collection.nextPage()
            }
        }
    }
