func messageCreateFile() {
        let channelId = "channelId"
        let fileURL = URL(fileURLWithPath: "/path/to/document.pdf")
        let fileData = try! Data(contentsOf: fileURL)
        
        let repository = AmityMessageRepository(client: AmityManager.shared.client!)
        let builder = AmityFileMessageCreationBuilder()
        builder.setFile(fileData, fileName: "document.pdf", mimeType: "application/pdf")
        
        repository.createMessage(withChannelId: channelId, builder: builder) { message, error in
            if let error = error {
                print("Error creating file message: \(error)")
                return
            }
            print("File message created: \(message?.messageId ?? "")")
        }
    }
