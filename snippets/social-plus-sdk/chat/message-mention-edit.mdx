fun editMessageWithMention(
        messageRepository: AmityMessageRepository,
        message: AmityMessage,
        mentionedUsers: List<AmityUser>
    ) {
        // mention specific users
        val mentionUserIds = mentionedUsers.map {
            it.getUserId()
        }

        val messageData = message.getData()
        if (messageData is AmityMessage.Data.TEXT) {
            messageRepository.editTextMessage(
                messageId = message.getMessageId()
            )
                .text(text = "hi @Richard, do you have a second?")
                .metadata(metadata = JsonObject()) // flexible data structure for highlighting text
                .mentionUsers(userIds = mentionUserIds)
                .build()
                .apply()
                .doOnComplete {
                    //success
                }
                .doOnError {
                    // Exception
                }
                .subscribe()
        }
    }
