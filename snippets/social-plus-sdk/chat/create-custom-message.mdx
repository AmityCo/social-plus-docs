import { MessageRepository } from '@amityco/ts-sdk'

async function createCustomMessage(channelId: string, customData: Record<string, any>) {
  try {
    const { data: message } = await MessageRepository.createMessage({
      channelId,
      dataType: 'custom',
      data: customData,
      metadata: { version: '1.0' },
    });
    console.log('Custom message created:', message.messageId, 'dataType:', message.dataType);
    return message;
  } catch (error) {
    console.error('Failed to create custom message:', error);
  }
}
