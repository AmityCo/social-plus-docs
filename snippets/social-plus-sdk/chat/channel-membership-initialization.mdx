import { ChannelRepository } from '@amityco/ts-sdk'

function observeChannelMembership(channelId: string) {
  const liveObject = ChannelRepository.getMembership(channelId, ({ data: membership, loading, error }) => {
    if (loading) return
    if (error) {
      console.error('Error observing membership:', error)
      return
    }

    console.log('Is member:', membership.isMember)
    console.log('Is banned:', membership.isBanned)
    console.log('Is muted:', membership.isMuted)
    console.log('Roles:', membership.roles)
  })

  return liveObject
}
