async function getVideoMessageAttribute() {
  const messageId = 'message-id-123'
  const liveObject = MessageRepository.getMessage(messageId)
  const unsubscribe = liveObject.subscribe(({ data: message, loading }) => {
    if (!loading && message && message.dataType === 'video') {
      const videoAttr = message.data as { fileId: string; thumbnailFileId: string }
      console.log('Video file ID:', videoAttr.fileId)
      console.log('Thumbnail file ID:', videoAttr.thumbnailFileId)
    }
  })
  return unsubscribe
}
