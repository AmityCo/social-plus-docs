import { MessageRepository } from '@amityco/ts-sdk'

async function createImageMessage(channelId: string, fileId: string, caption?: string) {
  try {
    const { data: message } = await MessageRepository.createMessage({
      channelId,
      dataType: 'image',
      data: { fileId, caption: caption || '' },
    });
    console.log('Image message created:', message.messageId);
    return message;
  } catch (error) {
    console.error('Failed to create image message:', error);
  }
}
