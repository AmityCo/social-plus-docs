fun filterChannel(channelRepository: AmityChannelRepository) {
        val includingTags = AmityTags().apply {
            add("games")
        }
        val excludingTags = AmityTags().apply {
            add("staff-only")
        }
        // Query for Community channels where the user is not member,
        // tagged as "games", and not tagged as "staff-only"
        channelRepository.getChannels()
            .communityType()
            .filter(filter = AmityChannelFilter.MEMBER)
