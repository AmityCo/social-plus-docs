import { MessageRepository } from '@amityco/ts-sdk'

async function createFileMessage(channelId: string, fileId: string, fileName: string) {
  try {
    const { data: message } = await MessageRepository.createMessage({
      channelId,
      dataType: 'file',
      data: { fileId },
      metadata: { fileName },
    });
    console.log('File message created:', message.messageId);
    return message;
  } catch (error) {
    console.error('Failed to create file message:', error);
  }
}
