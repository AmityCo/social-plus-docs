        let client = AmityClient.shared
        let messageRepo = AmityMessageRepository(client: client)
        let fileRepo = AmityFileRepository(client: client)
        var uploadedImages: [AmityImageData] = []
        let imageDataList = [Data(), Data()] // Your image data array
        let group = DispatchGroup()
        for imageData in imageDataList {
            group.enter()
            fileRepo.uploadImage(imageData, progress: nil) { imageFileData, error in
                if let imageFileData = imageFileData {
                    uploadedImages.append(imageFileData)
                }
                group.leave()
            }
        }
        group.notify(queue: .main) {
            let builder = AmityMessageCreationBuilder()
            builder.setSubChannelId("subChannelId")
            builder.setImageFileDataList(uploadedImages)
            messageRepo.createMessage(with: builder) { message, error in
                if let message = message {
                    print("Multi-image message created: \(message.messageId)")
                }
            }
        }
    }
