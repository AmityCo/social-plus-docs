fun createCustomMessage(
        messageRepository: AmityMessageRepository
    ) {
        val customObject = JsonObject()
        customObject.addProperty("url", "https://docs.amity.co")
        
        messageRepository.createCustomMessage(
            subChannelId = "subChannelId",
            data = customObject
        )
            .build()
            .send()
            .doOnComplete {
                // Void
            }
            .doOnError {
                // Exception
            }
            .subscribe()
    
        // Create custom message with metadata
        messageRepository.createCustomMessage(
                subChannelId = "subChannelId",
                data = customObject
        )
                .metadata(metadata = JsonObject()) // flexible data structure for additional custom data
                .build()
                .send()
                .doOnComplete {
                    // Void
                }
                .doOnError {
                    // Exception
                }
                .subscribe()
    }
