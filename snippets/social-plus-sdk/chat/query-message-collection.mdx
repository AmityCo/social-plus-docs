func queryMessageCollection() {
        let client = AmityClient.shared
        let messageRepository = AmityMessageRepository(client: client)
        let queryOptions = AmityMessageQueryOptions(
            subChannelId: "sub_channel_id",
            filterByParentId: false,
            parentId: nil,
            orderBy: .descending,
            includeDeleted: false
        )
        let liveCollection = messageRepository.getMessages(queryOptions)
        liveCollection.observe { collection, _, error in
            guard error == nil else { return }
            print("Messages: \(collection.count())")
        }
    }
