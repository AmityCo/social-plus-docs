import { ChannelRepository } from '@amityco/ts-sdk'

async function channelParticipationInit(channelId: string) {
  try {
    await ChannelRepository.joinChannel(channelId);
    console.log('Joined channel:', channelId);
    const liveObject = ChannelRepository.getChannel(channelId, ({ data: channel, loading, error }) => {
      if (!loading && channel) {
        console.log('Channel ready:', channel.channelId, 'Members:', channel.memberCount);
      }
    });
    return liveObject;
  } catch (error) {
    console.error('Failed to initialize channel participation:', error);
  }
}
