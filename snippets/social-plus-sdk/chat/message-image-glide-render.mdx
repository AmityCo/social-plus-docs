fun viewImageMessage(context: Context, imageView: ImageView, message: AmityMessage) {
        when (val data = message.getData()) {
            is AmityMessage.Data.IMAGE -> {
                val imageUrl: String? = data.getImage()?.getUrl()
                Glide.with(context)
                    .load(imageUrl)
                    .into(imageView)
            }
            else -> {}
        }
    }
