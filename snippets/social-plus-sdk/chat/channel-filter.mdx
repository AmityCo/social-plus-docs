import { ChannelRepository } from '@amityco/ts-sdk'

function queryChannelsByTypeAndMembership() {
  const liveCollection = ChannelRepository.queryChannels({
    types: ['community'],
    membership: 'member',
  })

  liveCollection.on('dataUpdated', (channels) => {
    console.log('Filtered channels:', channels.length)
    channels.forEach((channel) => {
      console.log(' -', channel.channelId, channel.displayName)
    })
  })

  liveCollection.on('dataError', (error) => {
    console.error('Error querying channels:', error)
  })
}
