import { MessageRepository } from '@amityco/ts-sdk'

async function createAudioMessage(channelId: string, fileId: string, duration: number) {
  try {
    const { data: message } = await MessageRepository.createMessage({
      channelId,
      dataType: 'audio',
      data: { fileId },
      metadata: { duration },
    });
    console.log('Audio message sent:', message.messageId);
    return message;
  } catch (error) {
    console.error('Failed to create audio message:', error);
  }
}
