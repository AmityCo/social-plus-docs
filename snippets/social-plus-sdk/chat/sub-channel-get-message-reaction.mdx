import { ReactionRepository } from '@amityco/ts-sdk'

function getMessageReactions() {
  const liveCollection = ReactionRepository.queryReactions({
    referenceId: 'messageId',
    referenceType: 'message',
  })

  liveCollection.on('dataUpdated', (reactions) => {
    console.log('Message reactions:', reactions)
  })

  liveCollection.on('dataError', (error) => {
    console.error('Failed to get message reactions:', error)
  })

  return () => liveCollection.dispose()
}
