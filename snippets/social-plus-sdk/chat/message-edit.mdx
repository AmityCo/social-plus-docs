fun editMessage(
        messageRepository: AmityMessageRepository,
        message: AmityMessage
    ) {
        when (val data = message.getData()) {
            is AmityMessage.Data.TEXT -> {
                messageRepository.editTextMessage(messageId = message.getMessageId())
                    .text(text = "text to edit")
                    .build()
                    .apply()
                    .doOnComplete {
                        //success
                    }
                    .subscribe()
            }
            is AmityMessage.Data.CUSTOM -> {
                messageRepository.editCustomMessage(messageId = message.getMessageId())
                    .data(data = JsonObject())
                    .build()
                    .apply()
                    .doOnComplete {
                        //success
                    }
                    .subscribe()
            }
            else -> {}
        }
    }
