import { MessageRepository } from '@amityco/ts-sdk'

async function messageReply(channelId: string, parentMessageId: string, text: string) {
  const message = await MessageRepository.createMessage({
    channelId,
    dataType: 'text',
    data: { text },
    parentId: parentMessageId,
  })
  console.log('Reply sent:', message.messageId)
  return message
}
