func messageSendFile() {
        let messageRepository = AmityMessageRepository(client: AmityManager.shared.client!)
        guard let fileUrl = Bundle.main.url(forResource: "document", withExtension: "pdf") else { return }
        let builder = AmityFileMessageBuilder(fileUrl: fileUrl)
        messageRepository.createMessage(
            withChannelId: "channelId123",
            builder: builder
        ) { message, error in
            guard let message = message else { return }
            print("File message sent: \(message.messageId)")
        }
    }

async function messageSendFile(channelId: string, fileId: string, fileName: string) {
