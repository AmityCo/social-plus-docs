import { MessageRepository } from '@amityco/ts-sdk'

async function messageCreateText(subChannelId: string, text: string) {
  const message = await MessageRepository.createMessage({
    subChannelId,
    dataType: 'text',
    data: { text },
  })
  console.log('Text message created:', message.data.messageId)
}
