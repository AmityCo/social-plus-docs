fun createChannelRepositoryUsage() {
        AmityChatClient.newChannelRepository()
            .createChannel()
            .community("channelId")
            .userIds(listOf("userId1", "userId2"))
            .displayName("Community Channel")
            .build()
            .create()
            .subscribeOn(io.reactivex.rxjava3.schedulers.Schedulers.io())
            .observeOn(AndroidSchedulers.mainThread())
            .subscribe({ channel ->
                val channelId = channel.getChannelId()
            }, { error ->
                error.printStackTrace()
            })
    }

func createChannelRepositoryUsage() {
        let channelRepository = AmityChannelRepository(client: AmityManager.shared.client!)
        let builder = AmityConversationChannelBuilder()
        builder.setUserId("userId")
        channelRepository.createChannel(with: builder) { channel, error in
            if let error = error {
                print("Failed to create channel: \(error)")
                return
            }
            print("Channel created: \(channel?.channelId ?? "")")
        }
