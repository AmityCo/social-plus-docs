import { ChannelRepository } from '@amityco/ts-sdk'

async function createChannel(channelId: string, displayName: string) {
  try {
    const { data: channel } = await ChannelRepository.createChannel({
      channelId,
      displayName,
      type: 'community',
      tags: ['public'],
      metadata: { topic: 'general' },
    });
