import { MessageRepository } from '@amityco/ts-sdk'

async function sendTextMessage(subChannelId: string, text: string) {
  const message = await MessageRepository.createMessage({
    subChannelId,
    dataType: 'text',
    data: { text },
  })

  console.log('Text message sent:', message.messageId)
  return message
}
