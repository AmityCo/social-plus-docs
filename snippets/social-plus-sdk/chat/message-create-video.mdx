func messageCreateVideo() {
        let channelId = "channelId"
        let videoURL = URL(fileURLWithPath: "/path/to/video.mp4")
        let videoData = try! Data(contentsOf: videoURL)
        
        let repository = AmityMessageRepository(client: AmityManager.shared.client!)
        let builder = AmityVideoMessageCreationBuilder()
        builder.setVideo(videoData, fileName: "video.mp4", mimeType: "video/mp4")
        
        repository.createMessage(withChannelId: channelId, builder: builder) { message, error in
            if let error = error {
                print("Error creating video message: \(error)")
                return
            }
            print("Video message created: \(message?.messageId ?? "")")
        }
    }
