import { ChannelRepository } from '@amityco/ts-sdk'

function observeChannel(channelId: string) {
  const unsubscribe = ChannelRepository.getChannel(channelId, ({ data: channel, loading, error }) => {
    if (loading) return
    if (error) {
      console.error('Error fetching channel:', error)
      return
    }

    console.log('Channel display name:', channel.displayName)
    console.log('Channel type:', channel.type)
    console.log('Member count:', channel.memberCount)
  })

  return unsubscribe
}
