func messageCreateAudio() {
        let channelId = "channelId"
        let audioFileURL = URL(fileURLWithPath: "/path/to/audio.m4a")
        let audioData = try! Data(contentsOf: audioFileURL)
        
        let repository = AmityMessageRepository(client: AmityManager.shared.client!)
        let builder = AmityAudioMessageCreationBuilder()
        builder.setAudio(audioData, fileName: "audio.m4a", mimeType: "audio/m4a")
        
        repository.createMessage(withChannelId: channelId, builder: builder) { message, error in
            if let error = error {
                print("Error creating audio message: \(error)")
                return
            }
            print("Audio message created: \(message?.messageId ?? "")")
        }
    }
