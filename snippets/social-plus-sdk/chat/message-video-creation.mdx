import { MessageRepository, FileRepository } from '@amityco/ts-sdk'

async function createVideoMessage() {
  const { data: uploadedFile } = await FileRepository.uploadFile({
    file: new File([], 'video.mp4', { type: 'video/mp4' }),
  })

  const { data: message } = await MessageRepository.createMessage({
    subChannelId: 'subChannelId',
    dataType: 'video',
    data: { fileId: uploadedFile.fileId },
  })
  console.log(message)
}
