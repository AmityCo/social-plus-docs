fun playVideoInResolution(context: Context, video: AmityVideo, resolution: AmityVideoResolution) {
		val url = video.getVideoUrl(resolution)
		
		val playVideo = Intent(Intent.ACTION_VIEW)
		playVideo.setDataAndType(Uri.parse(url), "video/mp4")
