function channelMemberPreview(channelId: string) {
  const liveCollection = ChannelRepository.Membership.getMembers(
    { channelId, limit: 5 },
    ({ data: members, loading, error }) => {
      if (error) { console.error(error); return; }
      if (!loading) {
        console.log('Channel members preview:');
        members.forEach(m => console.log('-', m.userId, m.displayName));
      }
    }
  );
  return liveCollection;
}
