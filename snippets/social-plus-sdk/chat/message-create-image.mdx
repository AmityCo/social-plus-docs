func messageCreateImage() {
        let channelId = "channelId"
        let image = UIImage(named: "photo")!
        let imageData = image.jpegData(compressionQuality: 0.8)!
        
        let repository = AmityMessageRepository(client: AmityManager.shared.client!)
        let builder = AmityImageMessageCreationBuilder()
        builder.setImage(imageData, fileName: "photo.jpg", mimeType: "image/jpeg")
        
        repository.createMessage(withChannelId: channelId, builder: builder) { message, error in
            if let error = error {
                print("Error creating image message: \(error)")
                return
            }
            print("Image message created: \(message?.messageId ?? "")")
        }
    }
