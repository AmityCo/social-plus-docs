```typescript TypeScript
import { ChannelRepository } from '@amityco/ts-sdk'

function listChannelMembers(channelId: string) {
  const liveCollection = ChannelRepository.queryMembers({ channelId })

  liveCollection.on('dataUpdated', (members) => {
    console.log(`Channel ${channelId} has ${members.length} member(s):`)
    members.forEach((member) => {
      console.log(' -', member.userId, '| role:', member.roles?.join(', '))
    })
  })

  liveCollection.on('dataError', (error) => {
    console.error('Error listing channel members:', error)
  })
}
```
