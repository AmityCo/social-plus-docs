import { ChannelRepository } from '@amityco/ts-sdk'

function getChannelMembers() {
  const liveCollection = ChannelRepository.queryMembers({
    channelId: 'channelId',
    membership: 'all',
  })

  liveCollection.on('dataUpdated', (members) => {
    console.log('Channel members:', members)
  })

  liveCollection.on('dataError', (error) => {
    console.error('Failed to get channel members:', error)
  })

  return () => liveCollection.dispose()
}
