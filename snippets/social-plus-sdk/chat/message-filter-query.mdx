fun filterMessages(
        messageRepository: AmityMessageRepository,
        parentMessage: AmityMessage
    ) {
        //including tags
        val includingTags = AmityTags().apply {
            add("games")
        }
        //excluding tags
        val excludingTags = AmityTags().apply {
            add("staff-only")
        }
        //parent messageId
        val parentId = parentMessage.getMessageId()

        //type
        val type = AmityMessage.DataType.IMAGE
        // queries for children messages of a message with id "m50219a" and tagged as "games"
        // excluding "staff-only" tag also get only image type messages
        messageRepository.getMessages(subChannelId = "subChannelId")
            .parentId(parentId = parentId) // optional
            .includingTags(includingTags = includingTags) // including "games" tag
            .excludingTags(excludingTags = excludingTags) // excluding "staff-only" tag
            .type(type = type) // optional
            .build()
            .query()
            .doOnNext {
                // PagingData<AmityMessage>
            }
            .doOnError {
                // Exception
            }
            .subscribe()
    }
