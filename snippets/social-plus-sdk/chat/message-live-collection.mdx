  void paginationListener() {
    //check if

    //#2 live collection has next page to load more
    if ((scrollcontroller.position.pixels >=

import { MessageRepository } from '@amityco/ts-sdk'

function observeMessages() {
  const liveCollection = MessageRepository.queryMessages({
    subChannelId: 'subChannelId',
    stackFromEnd: true,
  })

  liveCollection.on('dataUpdated', (messages) => {
    console.log('Messages updated:', messages)
  })

  liveCollection.on('loadingStatusChanged', (loading) => {
    console.log('Loading:', loading)
  })

  liveCollection.nextPage()
  return liveCollection
}
