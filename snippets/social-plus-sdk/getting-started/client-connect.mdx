import { Client } from '@amityco/ts-sdk'

async function connectClient(apiKey: string, userId: string, displayName: string) {
  try {
    Client.setup({ apiKey });
    const sessionHandler = {
      sessionWillRenewAccessToken(renewal) {
        renewal.renew(userId, 'your-auth-token');
      },
    };
    await Client.login({ userId, displayName }, sessionHandler);
    console.log('Client connected as:', userId);
  } catch (error) {
    console.error('Failed to connect client:', error);
  }
}
