fun setAccessTokenHandler(userId: String) {
        val sessionHandler = object : SessionHandler {
            override fun sessionWillRenewAccessToken(renewal: AccessTokenRenewal) {
                // Fetch a new auth token from your server and call renewWithAuthToken
                renewal.renewWithAuthToken(authToken = "newAuthToken")
            }
        }
        AmityCoreClient.login(userId = userId, sessionHandler = sessionHandler)
            .build()
            .submit()
            .subscribe()
    }
