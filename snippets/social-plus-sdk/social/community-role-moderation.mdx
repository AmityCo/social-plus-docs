fun communityRoleModeration(communityId: String, userId: String, role: String) {
        AmitySocialClient.newCommunityRepository()
            .moderation(communityId = communityId)
            .addRoles(roles = listOf(role), userIds = listOf(userId))
            .subscribe(
                { println("Role $role assigned to $userId") },
                { error -> println("Role assignment failed: ${error.message}") }
            )
    }
