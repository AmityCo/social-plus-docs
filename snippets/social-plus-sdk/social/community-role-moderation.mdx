import { CommunityRepository } from '@amityco/ts-sdk'

async function communityRoleModeration(communityId: string, userId: string, action: 'addModerator' | 'removeModerator') {
  try {
    if (action === 'addModerator') {
      await CommunityRepository.Moderation.addRole(communityId, 'community-moderator', [userId]);
      console.log('User promoted to moderator:', userId);
    } else {
      await CommunityRepository.Moderation.removeRole(communityId, 'community-moderator', [userId]);
      console.log('User removed from moderator role:', userId);
    }
  } catch (error) {
    console.error('Failed to update community role:', error);
  }
}
