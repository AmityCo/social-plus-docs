fun renderPostImageByGlide(
        context: Context,
        imageView: ImageView,
        post: AmityPost
    ) {
        val data = post.getData()
        when (data) {
            is AmityPost.Data.IMAGE -> {
                val image = data.getImage()
                val imageUrl = image?.getUrl()
                Glide.with(context)
                    .load(imageUrl)
                    .into(imageView)
            }
            else -> {}
        }
    }
