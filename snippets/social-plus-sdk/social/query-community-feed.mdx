import { PostRepository } from '@amityco/ts-sdk'

function queryCommunityFeed() {
  const liveCollection = PostRepository.getPosts({
    targetType: 'community',
    targetId: 'communityId',
    sortBy: 'lastCreated',
    includeDeleted: false,
  })

  liveCollection.on('dataUpdated', (posts) => {
    console.log('Community feed posts:', posts)
  })

  liveCollection.on('dataError', (error) => {
    console.error('Failed to query community feed:', error)
  })

  return () => liveCollection.dispose()
}
