import { UserRepository } from '@amityco/ts-sdk'

function queryOtherUserFollowings(userId: string) {
  const liveCollection = UserRepository.getFollowings(userId)

  liveCollection.on('dataUpdated', (followings) => {
    console.log(`Followings of ${userId}:`, followings.length)
    followings.forEach((following) => {
      console.log('Following:', following.userId)
    })
  })

  liveCollection.on('dataError', (error) => {
    console.error('Error querying followings:', error)
  })

  return () => liveCollection.dispose()
}
