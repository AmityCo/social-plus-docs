import { FeedRepository } from '@amityco/ts-sdk'

async function globalFeedPostTypes() {
  const allowedPostTypes = ['text', 'image', 'video', 'liveStream']
  const liveCollection = FeedRepository.getGlobalFeed({
    postTypes: allowedPostTypes,
    limit: 20,
  })
  const unsubscribe = liveCollection.subscribe(({ data: posts, loading }) => {
    if (!loading) console.log('Global feed posts:', posts)
  })
  return unsubscribe
}
