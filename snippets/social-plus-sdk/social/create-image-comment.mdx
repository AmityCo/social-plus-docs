async function createImageComment(referenceId: string, text: string, imageFileId: string) {
  try {
    const { data: comment } = await CommentRepository.createComment({
      referenceId,
      referenceType: 'post',
      data: { text },
      attachments: [{ fileId: imageFileId, type: 'image' }],
    });
