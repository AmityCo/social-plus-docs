import { PostRepository } from '@amityco/ts-sdk'

async function hashtagQueryFeed(hashtag: string) {
  const { data: posts, onNextPage } = await PostRepository.getPosts({
    targetType: 'community',
    targetId: hashtag,
    tags: [hashtag],
  })
  console.log(`Posts with hashtag #${hashtag}:`, posts)
  return { posts, onNextPage }
}
