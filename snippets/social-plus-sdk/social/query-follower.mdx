async function queryFollowers() {
  const params = {
    userId: 'userId123',
    limit: 20,
  };

  const liveCollection = FollowRepository.getFollowers(params);
  liveCollection.on('dataUpdated', followers => {
    console.log('Followers:', followers.map(f => f.userId));
  });
