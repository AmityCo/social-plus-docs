import { StoryRepository } from '@amityco/ts-sdk'

async function queryStoryTargets() {
  const params = {
    targets: [{ targetId: 'communityId123', targetType: 'community' as const }],
  };

  const liveCollection = StoryRepository.getStoryTargets(params);
  liveCollection.on('dataUpdated', targets => {
    console.log('Story targets:', targets.map(t => t.targetId));
  });
  liveCollection.on('dataError', err => console.error('Error querying story targets:', err));
}
