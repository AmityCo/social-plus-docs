fun renderPostImageByPicasso(
        context: Context,
        imageView: ImageView,
        post: AmityPost
    ) {
        val data = post.getData()
        when (data) {
            is AmityPost.Data.IMAGE -> {
                val image = data.getImage()
                val imageUrl = image?.getUrl()
                val picasso = Picasso.Builder(context).build()
                picasso.load(imageUrl)
                    .into(imageView)
            }
            else -> {}
        }
    }
