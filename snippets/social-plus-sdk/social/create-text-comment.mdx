import { CommentRepository } from '@amityco/ts-sdk'

async function createTextComment(postId: string, text: string) {
  const { data: comment } = await CommentRepository.createComment({
    referenceType: 'post',
    referenceId: postId,
    data: { text },
  })

  console.log('Comment created:', comment.commentId)
  return comment
}

// Example usage
createTextComment('postId', 'This is a great post!')
