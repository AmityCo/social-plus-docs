import { CommunityRepository } from '@amityco/ts-sdk'

async function leaveCommunity() {
  const communityId = 'community-id-123'

  // Leave the community first
  const { error: leaveError } = await CommunityRepository.leaveCommunity(communityId)
  if (leaveError) {
    console.error('Failed to leave community:', leaveError)
    return
  }
  console.log('Left community successfully')

  // Optionally delete if owner
  const { error: deleteError } = await CommunityRepository.deleteCommunity(communityId)
  if (deleteError) {
    console.error('Failed to delete community:', deleteError)
  } else {
    console.log('Community deleted successfully')
  }
}
