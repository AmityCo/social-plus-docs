fun createPost(communityId: String, text: String) {
        AmitySocialClient.newPostRepository()
            .createPost()
            .targetCommunity(communityId)
            .text(text)
            .build()
            .post()
            .subscribe(
                { post -> println("Post created: ${post.getPostId()}") },
                { error -> println("Create failed: ${error.message}") }
            )
    }
