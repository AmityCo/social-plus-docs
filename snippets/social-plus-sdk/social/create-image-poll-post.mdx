import { PostRepository } from '@amityco/ts-sdk'

async function createImagePollPost(targetId: string, question: string, answers: string[], imageFileId: string) {
  try {
    const { data: poll } = await PollRepository.createPoll({
      question,
      answers: answers.map(text => ({ dataType: 'text', data: { text } })),
      timeToClosePoll: 86400,
      answerType: 'single',
    });
    const { data: post } = await PostRepository.createPost({
      targetType: 'community',
      targetId,
      dataType: 'poll',
      data: { pollId: poll.pollId },
      attachments: [{ fileId: imageFileId, type: 'image' }],
    });
    console.log('Image poll post created:', post.postId, 'pollId:', poll.pollId);
    return post;
  } catch (error) {
    console.error('Failed to create image poll post:', error);
  }
}
