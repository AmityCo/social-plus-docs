import { PostRepository, FileRepository } from '@amityco/ts-sdk'

async function createAudioPost(audioFile) {
  const { data: uploadedFile } = await FileRepository.uploadAudio(audioFile)
  const { data: post } = await PostRepository.createPost({
    targetType: 'community',
    targetId: 'communityId',
    dataType: 'audio',
    data: { fileId: uploadedFile.fileId },
  })
  console.log(post)
}
