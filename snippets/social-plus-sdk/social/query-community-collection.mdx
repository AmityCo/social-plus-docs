func queryCommunityCollection() {
        let client = AmityClient.shared
        let communityRepository = AmityCommunityRepository(client: client)
        let query = AmityCommunityQuery()
        query.filter = .all
        query.sortBy = .displayName
        let liveCollection = communityRepository.getCommunities(query)
        liveCollection.observe { collection, _, error in
            guard error == nil else { return }
            print("Communities: \(collection.count())")
        }
    }
