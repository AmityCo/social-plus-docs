import { UserRepository } from '@amityco/ts-sdk'

function getPendingFollowers(myUserId: string) {
  const liveCollection = UserRepository.getFollowers(myUserId, { status: 'pending' })

  liveCollection.on('dataUpdated', (followers) => {
    console.log('Pending follow requests:', followers)
  })

  liveCollection.on('dataError', (error) => {
    console.error('Error fetching pending followers:', error)
  })

  return () => liveCollection.dispose()
}
