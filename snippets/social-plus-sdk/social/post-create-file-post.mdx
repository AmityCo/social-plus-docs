void createFilePost(
      String targetId, AmityPostTargetType targetType, List<File> files) async {
    final uploadedFiles = <AmityFile>[];

    for (final file in files) {
      await AmityCoreClient.newFileRepository()
          .uploadFile(file)
          .stream
          .firstWhere((result) => result is AmityUploadComplete)
          .then((result) {
        result.when(
          complete: (f) => uploadedFiles.add(f),
          progress: (_, __) {},
          error: (_) {},
          cancel: () {},
        );
      });
    }

    AmitySocialClient.newPostRepository()
        .createPost()
        .targetType(targetType)
        .targetId(targetId)
