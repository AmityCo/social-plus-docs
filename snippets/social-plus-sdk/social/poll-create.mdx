```swift iOS
func pollCreate() {
        let pollRepository = AmityPollRepository(client: AmityClient.shared)
        let builder = AmityPollCreationBuilder()
        builder.setQuestion("What is your favourite color?")
        builder.setAnswers(["Red", "Green", "Blue"])
        builder.setMultipleSelection(false)
        builder.setClosedIn(86400)
        pollRepository.createPoll(builder) { poll, error in
            if let error = error {
                print("Failed to create poll: \(error.localizedDescription)")
                return
            }
            print("Poll created with ID: \(poll?.pollId ?? "")")
        }
    }
```
