async function createVideoStory(communityId: string, videoFile: File) {
  const formData = new FormData()
  formData.append('file', videoFile)

  const metadata = {
    url: 'https://example.com',
  }

  const { data: story } = await StoryRepository.createVideoStory(
    'community',
    communityId,
    formData,
    metadata,
  )

  console.log('Video story created:', story.storyId)
  return story
