    // Flutter equivalent of a live collection using PagingController
    final _posts = <AmityPost>[];
    late PagingController<AmityPost> _controller;

    _controller = PagingController(
      pageFuture: (token) => AmitySocialClient.newFeedRepository()
          .getCommunityFeed(communityId: 'communityId')
          .getPagingData(token: token, limit: 20),
      pageSize: 20,
    )..addListener(() {
        if (_controller.error == null) {
          _posts
            ..clear()
            ..addAll(_controller.loadedItems);
