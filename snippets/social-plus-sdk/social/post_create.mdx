func postCreate() {
        let postRepository = AmityPostRepository(client: AmityClient.shared)
        let builder = AmityPostCreationBuilder()
        builder.setTargetType(.community, targetId: "communityId")
        builder.setText("Hello, Social Plus!")
        postRepository.createPost(builder) { post, error in
            if let error = error {
                print("Failed to create post: \(error.localizedDescription)")
                return
            }
            print("Post created with ID: \(post?.postId ?? "")")
        }
    }
