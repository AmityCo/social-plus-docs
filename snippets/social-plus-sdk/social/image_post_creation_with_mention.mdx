import { FileRepository, PostRepository } from '@amityco/ts-sdk'

async function imagePostCreationWithMention() {
  const communityId = 'community-id-123'
  const fileInput = document.getElementById('fileInput') as HTMLInputElement
  const file = fileInput.files?.[0]
  if (!file) return

  const uploadResult = await FileRepository.uploadImage(file)
  const { data: post, error } = await PostRepository.createPost({
    targetType: 'community',
    targetId: communityId,
    dataType: 'image',
    data: { fileId: uploadResult.data?.fileId ?? '' },
    text: 'Hello @john, check this out!',
    mentionees: [{ type: 'user', userIds: ['user-id-john'] }],
    metadata: {
      mentioned: [{ type: 'user', userId: 'user-id-john', index: 6, length: 5 }],
    },
  })
  if (error) {
    console.error('Failed to create image post with mention:', error)
  } else {
    console.log('Image post with mention created:', post?.postId)
  }
}
