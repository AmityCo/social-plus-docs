```typescript TypeScript
import { StoryRepository } from '@amityco/ts-sdk'

function getGlobalStoryTargets() {
  const unsubscribe = StoryRepository.getGlobalStoryTargets(
    { limit: 10 },
    ({ data: targets, loading, error }) => {
      if (loading) return
      if (error) {
        console.error('Error fetching global story targets:', error)
        return
      }

      console.log('Global story targets:', targets.length)
      targets.forEach((target) => {
        console.log('Target ID:', target.targetId, '| Has stories:', target.hasStories)
      })
    },
  )

  return unsubscribe
}
```
