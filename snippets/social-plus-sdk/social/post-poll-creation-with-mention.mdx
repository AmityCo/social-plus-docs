        // To create a post on a community feed, create a community target
        val communityTarget = AmityPost.Target.COMMUNITY.create(communityId = "community1")

        // text with mention
        val text = "What are your favorite songs @userId1 @userId2"

        // create user1 mention data
        val userId1 = "userId1"
        val startMentionIndex1 = 29
        val mentionLength1 = 7
        val mentionMetadata1: AmityMentionMetadata =
            AmityMentionMetadata.USER(
                userId = userId1,
                index = startMentionIndex1,
                length = mentionLength1
            )
        // create user2 mention data
        val userId2 = "userId2"
        val startMentionIndex2 = 38
        val mentionLength2 = 7
        val mentionMetadata2: AmityMentionMetadata =
            AmityMentionMetadata.USER(
                userId = userId2,
                index = startMentionIndex2,
                length = mentionLength2
            )
        // construct user1 and user2 data to JsonObject using AmityMentionMetadataCreator
        val mentionMetadataCreator: JsonObject =
            AmityMentionMetadataCreator(
                mentionMetaData = listOf(
                    mentionMetadata1,
                    mentionMetadata2
                )
            ).create()
        val mentionUserIds = setOf("userId1", "userId2")

        // create poll
        pollRepository.createPoll(question = text) // a question
            .answers(

                    AmityPollAnswer.Data.TEXT("All of the small things"),
                    AmityPollAnswer.Data.TEXT("The rock show")
                )
            )
            .answerType(answerType = AmityPoll.AnswerType.MULTIPLE) // allow users to vote on multiple choices.

            .create()

                // AmityPost
