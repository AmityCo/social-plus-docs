async function createCommentEditor(referenceId: string, text: string, mentionees?: any[]) {
  try {
    const { data: comment } = await CommentRepository.createComment({
      referenceId,
      referenceType: 'post',
      data: { text },
      mentionees: mentionees || [],
      metadata: { source: 'editor' },
    });
    console.log('Comment created via editor:', comment.commentId);
    return comment;

    console.error('Failed to create comment from editor:', error);
  }
}
