func postTextCreation() {
        let feedRepository = AmityFeedRepository(client: AmityManager.shared.client!)
        let builder = AmityTextPostBuilder()
        builder.setTargetType(.community)
        builder.setTargetId("communityId")
        builder.setText("Hello, community! This is my first post.")

            if let error = error {
                print("Text post creation failed: \(error)")
                return
            }
            print("Text post created: \(post?.postId ?? "")")
        }
