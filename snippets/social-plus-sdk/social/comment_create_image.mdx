func commentCreateImage() {
        let commentRepository = AmityCommentRepository(client: AmityManager.shared.client!)
        let image = UIImage(named: "photo")!
        let imageData = image.jpegData(compressionQuality: 0.8)!
        
        let fileRepository = AmityFileRepository(client: AmityManager.shared.client!)
        fileRepository.uploadImage(imageData, fileName: "photo.jpg", mimeType: "image/jpeg") { uploadedImage, error in
            guard let imageId = uploadedImage?.fileId else { return }
            
            let builder = AmityCommentCreationBuilder()
            builder.setText("Check out this image!")
            builder.setAttachments([AmityCommentAttachment(fileId: imageId, type: .image)])
            
            commentRepository.createComment(withBuilder: builder, referenceId: "postId", referenceType: .post) { comment, error in
                if let error = error {
                    print("Image comment creation failed: \(error)")
                    return
                }
                print("Image comment created: \(comment?.commentId ?? "")")
            }
        }
    }
