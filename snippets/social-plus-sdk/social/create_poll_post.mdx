        let question = "What is your favorite feature?"
        let answers = ["Live Objects", "Real-time Chat", "Communities"]

            guard let poll = poll, error == nil else { return }
            let builder = AmityPollPostBuilder()
            builder.setPollId(poll.pollId)

            }
        }
    }
