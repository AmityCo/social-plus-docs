fun uploadFile(
        fileRepository: AmityFileRepository,
        fileUri: Uri
    ) {
        fileRepository.uploadFile(uri = fileUri)
            .doOnNext { uploadResult: AmityUploadResult<AmityFile> ->
                when (uploadResult) {
                    is AmityUploadResult.PROGRESS -> {
                        // upload progress event
                        val progress = uploadResult.getUploadInfo().getProgressPercentage()
                    }
                    is AmityUploadResult.COMPLETE -> {
                        // completion event
                        val file: AmityFile = uploadResult.getFile()
                    }
                    is AmityUploadResult.ERROR -> {
                        // error event
                        val error = AmityError.from(uploadResult.getError())
                    }
                    is AmityUploadResult.CANCELLED -> {
                        // cancelled upload event
                    }
                }
            }
            .subscribe()
    }

import { PostRepository } from '@amityco/ts-sdk'

async function postFileUpload(targetId: string, targetType: string, fileId: string, caption?: string) {
  const post = await PostRepository.createPost({
    targetType: targetType as any,
    targetId,
    dataType: 'file',
    data: { text: caption ?? '' },
    attachments: [{ type: 'file', fileId }],
  })
  console.log('File post created:', post.postId)
  return post
}
