import { StoryRepository } from '@amityco/ts-sdk'

function getActiveStoryTargets() {
  const unsubscribe = StoryRepository.getGlobalStoryTargets(
    (response) => {
      if (response.error) throw response.error
      if (!response.loading) {
        console.log('Active story targets:', response.data)
        response.data.forEach((target) => {
          console.log('Target type:', target.targetType)
          console.log('Target ID:', target.targetId)
        })
      }
    },
  )
  return unsubscribe
}
