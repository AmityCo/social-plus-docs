        commentRepository: AmityCommentRepository,
        comment: AmityComment
    ) {
        val postId = (comment.getReference() as AmityComment.Reference.POST).getPostId()
        commentRepository.createComment()
            .post(postId = postId)
            .parentId(parentId = comment.getCommentId()) // id of the comment to be replied
            .with()
            .text(text = "reply to a comment!")
            .build()
            .send()
            .doOnSuccess { comment: AmityComment ->
                // AmityComment
            }
            .doOnError {
                // Exception
            }
            .subscribe()
    }
