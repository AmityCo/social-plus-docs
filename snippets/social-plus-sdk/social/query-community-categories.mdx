import { CommunityRepository } from '@amityco/ts-sdk'

async function queryCommunityCategories() {
  const params = {
    sortBy: 'name' as const,
    includeDeleted: false,
  };

  const liveCollection = CommunityRepository.getCategories(params);
  liveCollection.on('dataUpdated', categories => {
    console.log('Categories:', categories.map(cat => cat.name));
  });
  liveCollection.on('dataError', err => console.error('Error querying categories:', err));
}
