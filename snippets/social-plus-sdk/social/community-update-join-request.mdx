fun communityUpdateJoinRequest(communityId: String, userId: String, accept: Boolean) {
        val action = if (accept) {
            AmitySocialClient.newCommunityRepository()
                .acceptJoinRequest(communityId = communityId, userId = userId)
        } else {
            AmitySocialClient.newCommunityRepository()
                .declineJoinRequest(communityId = communityId, userId = userId)
        }
        action.subscribe(
            { println("Join request updated") },
            { error -> println("Error: ${error.message}") }
        )
    }
