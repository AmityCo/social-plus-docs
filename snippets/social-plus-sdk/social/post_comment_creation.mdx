import { CommentRepository } from '@amityco/ts-sdk'

async function createPostComment() {
  const commentData = {
    referenceId: 'postId123',
    referenceType: 'post' as const,
    data: { text: 'This is a great post!' },
  };

  const liveComment = CommentRepository.createComment(commentData);
  liveComment.on('dataUpdated', comment => console.log('Comment created:', comment.commentId));
  liveComment.on('dataError', err => console.error('Error creating comment:', err));
}
