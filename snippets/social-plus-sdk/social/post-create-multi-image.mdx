        ]
        var uploadedImages: [AmityImageData] = []
        let group = DispatchGroup()
        for url in imageURLs {
            group.enter()
            fileRepository.uploadImage(fromLocalURL: url, progress: nil) { imageData, _ in
                if let imageData = imageData { uploadedImages.append(imageData) }
                group.leave()
            }
        }
        group.notify(queue: .main) {
            let postRepository = AmityPostRepository(client: AmityClient.shared)
            let builder = AmityImagePostBuilder()
            builder.setImages(uploadedImages)
            postRepository.createPost(builder, targetId: nil, targetType: .user) { post, error in
                if let error = error {
                    print("Failed to create multi-image post: \(error.localizedDescription)")
                    return
                }
                print("Multi-image post created: \(post?.postId ?? "")")
