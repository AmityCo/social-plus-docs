import { StoryRepository } from '@amityco/ts-sdk'

function getStoryTarget(targetId: string, targetType: string) {
  const unsubscribe = StoryRepository.getTargetById(
    { targetId, targetType: targetType as 'community' },
    ({ data: target, loading, error }) => {
      if (loading) return
      if (error) {
        console.error('Error fetching story target:', error)
        return
      }

      console.log('Target ID:', target.targetId)
      console.log('Has stories:', target.hasStories)
    },
  )

  return unsubscribe
}
