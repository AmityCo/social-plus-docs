import { CommentRepository } from '@amityco/ts-sdk'

function getLatestComment(postId: string) {
  const liveCollection = CommentRepository.getComments(
    { referenceId: postId, referenceType: 'post', limit: 1, sortBy: 'lastCreated' },
  )

  liveCollection.on('dataUpdated', (comments) => {
    if (comments.length > 0) {
      const latestComment = comments[0]
      console.log('Latest comment ID:', latestComment.commentId)
      console.log('Latest comment text:', latestComment.data?.text)
    }
  })

  liveCollection.on('dataError', (error) => {
    console.error('Error fetching latest comment:', error)
  })

  return () => liveCollection.dispose()
}
