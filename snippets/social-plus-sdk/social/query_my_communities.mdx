import { CommunityRepository } from '@amityco/ts-sdk'

function queryMyCommunities() {
  const disposer = CommunityRepository.getCommunities(
    { membership: 'member', limit: 20 },
    ({ data: communities, loading, error }) => {
      if (loading) return
      if (error) {
        console.error('Failed to query communities', error)
        return
      }
      console.log('My communities:', communities)
    },
  )

  return disposer
}
