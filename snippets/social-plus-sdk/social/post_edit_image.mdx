import { PostRepository } from '@amityco/ts-sdk'

async function editImagePost() {
  const updateData = {
    postId: 'imagePostId123',
    data: { text: 'Updated caption for image post' },
    attachments: [{ type: 'image' as const, fileId: 'newImageFileId' }],
  };

  const livePost = PostRepository.updatePost(updateData);
  livePost.on('dataUpdated', post => console.log('Image post updated:', post.postId));

}
