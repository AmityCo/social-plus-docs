fun updateEvent(
        eventId: String,
        title: String,
        description: String,
        isInviteOnly: Boolean,
        startTime: DateTime,
        endTime: DateTime,
        location: String,
        externalUrl: String,
        coverImageFileId: String,
        tags: List<String>
    ) {
        val eventRepository = AmitySocialClient.newEventRepository()

        eventRepository.updateEvent(eventId)
            .setTitle(title)
            .setDescription(description)
            .setIsInviteOnly(isInviteOnly)
            .setStartTime(startTime)
            .setEndTime(endTime)
            .setLocation(location)
            .setExternalUrl(externalUrl)
            .setCoverImageFileId(coverImageFileId)
            .setTags(tags)
            .build()
            .apply()
            .subscribe(
                { updatedEvent: AmityEvent ->
                    // Event updated successfully
                    println("Event updated: ${updatedEvent.getTitle()}")
                },
                { error: Throwable ->
                    // Handle error
                    println("Error updating event: ${error.message}")
                }
            )
    }
