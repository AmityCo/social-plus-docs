void queryCommunityPosts() async {
    final postRepository = AmitySocialClient.newPostRepository();
    postRepository
        .getPosts()
        .targetCommunity('communityId')
        .sortBy(AmityPostSortOption.LAST_CREATED)
        .types([AmityDataType.TEXT, AmityDataType.IMAGE])
        .asStream()
        .listen((posts) {
      print('Posts: \${posts.length}');
    });
  }
