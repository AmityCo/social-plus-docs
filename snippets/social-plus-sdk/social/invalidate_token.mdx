async function invalidateToken() {
  const client = createClient('your-api-key')
  await client.login({ userId: 'user-id-123', displayName: 'John Doe' })

  // Force token invalidation to trigger re-authentication
  await client.logout()
  console.log('Token invalidated. Re-authenticating...')

  const { data: session, error } = await client.login({
    userId: 'user-id-123',
    displayName: 'John Doe',
    authToken: 'new-auth-token',
  })
  if (!error) console.log('Re-authenticated:', session?.userId)
}
