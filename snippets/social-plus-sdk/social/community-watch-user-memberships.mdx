import { CommunityRepository } from '@amityco/ts-sdk'

function watchUserMemberships(userId: string) {
  const liveCollection = CommunityRepository.Membership.getCommunityMembers(
    { userId, membership: 'all' },
    ({ data: memberships, loading, error }) => {
      if (error) { console.error(error); return; }
      if (!loading) {
        memberships.forEach(m => {
          console.log('Community:', m.communityId, 'Status:', m.membership);
        });
      }
    }
  );
  return liveCollection;
}
