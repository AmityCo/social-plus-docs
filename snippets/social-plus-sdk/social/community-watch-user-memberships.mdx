import { CommunityRepository } from '@amityco/ts-sdk'

function watchMyMemberships() {
  const liveCollection = CommunityRepository.getMyMemberships()

  liveCollection.on('dataUpdated', (memberships) => {
    console.log('My memberships:', memberships)
  })

  liveCollection.on('dataError', (error) => {
    console.error('Failed to watch memberships:', error)
  })

  return () => liveCollection.dispose()
}
