fun createImagePost(communityId: String, imageIds: List<String>, caption: String) {
        AmitySocialClient.newPostRepository()
            .createPost()
            .targetCommunity(communityId)
            .image(*imageIds.map { AmityImage(it) }.toTypedArray())
            .text(caption)
            .build()
            .post()
            .subscribe(
                { post -> println("Image post created: ${post.getPostId()}") },
                { error -> println("Create failed: ${error.message}") }
            )
    }

import { PostRepository } from '@amityco/ts-sdk'

async function createImagePost(targetId: string, imageFileIds: string[], caption?: string) {
  try {
    const { data: post } = await PostRepository.createPost({
      targetType: 'community',
      targetId,
      dataType: 'image',
      data: { fileId: imageFileIds[0] },
      attachments: imageFileIds.map(fileId => ({ fileId, type: 'image' })),
      text: caption || '',
    });
    console.log('Image post created:', post.postId);
    return post;
  } catch (error) {
    console.error('Failed to create image post:', error);
  }
}
