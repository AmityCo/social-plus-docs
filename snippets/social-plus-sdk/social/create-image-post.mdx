import { FileRepository, PostRepository } from '@amityco/ts-sdk'

async function createImagePost(imageFile: File, targetId: string) {
  const { data: uploadedImage } = await FileRepository.uploadImage(imageFile)

  const { data: post } = await PostRepository.createPost({
    targetId,
    targetType: 'community',
    dataType: 'image',
    data: { fileId: uploadedImage.fileId },
  })

  console.log('Image post created:', post.postId)
  return post
}
