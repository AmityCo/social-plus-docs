        let commentRepository = AmityCommentRepository(client: AmityManager.shared.client!)
        let builder = AmityCommentCreationBuilder()
        builder.setText("This is a great post!")
        
        commentRepository.createComment(withBuilder: builder, referenceId: "postId", referenceType: .post) { comment, error in
            if let error = error {
                print("Comment creation failed: \(error)")
                return
            }
            print("Comment created: \(comment?.commentId ?? "")")
        }
    }
