import { CommunityRepository } from '@amityco/ts-sdk'

function getCommunityMemberList() {
  const liveCollection = CommunityRepository.queryMembers({
    communityId: 'communityId',
    membership: 'all',
    sortBy: 'firstCreated',
  })

  liveCollection.on('dataUpdated', (members) => {
    console.log('Community members:', members)
  })

  liveCollection.on('dataError', (error) => {
    console.error('Failed to get member list:', error)
  })

  return () => liveCollection.dispose()
}
