import { PostRepository } from '@amityco/ts-sdk'

function viewPost(postId: string) {
  const unsubscribe = PostRepository.getPost(
    postId,
    ({ data: post, loading, error }) => {
      if (loading) return
      if (error) throw error
      console.log('Viewing post:', post.postId)
      console.log('View count:', post.viewCount)
      console.log('Reaction count:', post.reactionsCount)
    },
  )
  return unsubscribe
}
