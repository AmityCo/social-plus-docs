function getComment(commentId: string) {
  const unsubscribe = CommentRepository.getComment(
    commentId,
    ({ data: comment, loading, error }) => {
      if (loading) return
      if (error) throw error
      console.log('Comment text:', comment.data?.text)
    },
  )
  return unsubscribe
}
