import { CommunityRepository } from '@amityco/ts-sdk'

function searchCommunities(searchQuery: string) {
  const liveCollection = CommunityRepository.queryCommunities({
    displayName: searchQuery,
    membership: 'all',
  })

  liveCollection.on('dataUpdated', (communities) => {
    console.log('Search results:', communities)
  })

  liveCollection.on('dataError', (error) => {
    console.error('Community search error:', error)
  })

  return () => liveCollection.dispose()
}
