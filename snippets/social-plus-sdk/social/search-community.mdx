import { CommunityRepository } from '@amityco/ts-sdk'

async function searchCommunity() {
  const params = {
    displayName: 'tech',
    filter: 'all' as const,
    sortBy: 'displayName' as const,
  };

  const liveCollection = CommunityRepository.getCommunities(params);
  liveCollection.on('dataUpdated', communities => {
    console.log('Community search results:', communities.map(c => c.displayName));
  });
  liveCollection.on('dataError', err => console.error('Error searching communities:', err));
}
