import { FeedRepository } from '@amityco/ts-sdk'

function getCommunityFeed(communityId: string) {
  const unsubscribe = FeedRepository.getCommunityFeed(
    { communityId, feedType: 'published' },
    ({ data: posts, loading, error }) => {
      if (loading) return
      if (error) {
        console.error('Error fetching community feed:', error)
        return
      }

      console.log(`Posts in community ${communityId} feed:`, posts.length)
      posts.forEach((post) => {
        console.log(' -', post.postId, '|', post.data?.text ?? post.dataType)
      })
    },
  )

  return unsubscribe
}
