import { UserRepository } from '@amityco/ts-sdk'

function queryOtherUserFollowers(userId: string) {
  const liveCollection = UserRepository.getFollowers(userId)

  liveCollection.on('dataUpdated', (followers) => {
    console.log(`Followers of ${userId}:`, followers.length)
    followers.forEach((follower) => {
      console.log('Follower:', follower.userId)
    })
  })

  liveCollection.on('dataError', (error) => {
    console.error('Error querying followers:', error)
  })

  return () => liveCollection.dispose()
}
