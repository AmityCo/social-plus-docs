import { CommunityRepository } from '@amityco/ts-sdk'

async function createCommunity() {
  try {
    const { data: community } = await CommunityRepository.createCommunity({
      displayName: 'My New Community',
      description: 'A place for our group',
      isPublic: true,
      tags: ['tech', 'developers'],
      metadata: { category: 'technology' },
    });
    console.log('Community created:', community.communityId, community.displayName);
    return community;
  } catch (error) {
    console.error('Failed to create community:', error);
  }
}
