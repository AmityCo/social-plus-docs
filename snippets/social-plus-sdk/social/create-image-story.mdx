import { StoryRepository } from '@amityco/ts-sdk'

async function createImageStory(targetId: string, imageFileId: string, items?: any[]) {
  try {
    const { data: story } = await StoryRepository.createStory({
      targetType: 'community',
      targetId,
      dataType: 'image',
      data: { fileId: imageFileId },
      items: items || [],
    });
    console.log('Image story created:', story.storyId);
    return story;
  } catch (error) {
    console.error('Failed to create image story:', error);
  }
}
