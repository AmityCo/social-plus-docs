fun joinCommunity(
        communityRepository: AmityCommunityRepository,
        community: AmityCommunity
    ) {
        communityRepository
            .joinCommunity(communityId = community.getCommunityId())
            .doOnComplete {
                // Void
            }
            .doOnError { throwable: Throwable ->
                //Amity Error example
                when {
                    AmityError.from(throwable) == AmityError.ITEM_NOT_FOUND -> {
                        // Community doesn't exist
                    }
                    AmityError.from(throwable) == AmityError.USER_IS_BANNED -> {
                        // User is banned
                    }
                }
            }
            .subscribe()
    }
