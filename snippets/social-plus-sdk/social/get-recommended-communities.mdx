import { CommunityRepository } from '@amityco/ts-sdk'

function getRecommendedCommunities() {
  const disposer = CommunityRepository.getCommunities(
    { sortBy: 'lastCreated', isRecommended: true, limit: 20 },
    ({ data: communities, loading, error }) => {
      if (loading) return
      if (error) {
        console.error('Failed to fetch recommended communities', error)
        return
      }
      console.log('Recommended communities:', communities)
    },
  )

  return disposer
}
