import { FeedRepository, Client } from '@amityco/ts-sdk'

function getMyFeed() {
  const { userId } = Client.getActiveClient()

  const unsubscribe = FeedRepository.getUserFeed(
    { userId, feedType: 'published' },
    (response) => {
      if (response.error) throw response.error
      if (!response.loading) {
        console.log('My feed posts:', response.data)
      }
    },
  )
  return unsubscribe
}
