func getAttributedString(from post: AmityPost) -> NSAttributedString? {
    guard
        let metadata = post.metadata,
        let text = post.data?["text"] as? String else {
        return nil
    }
    // Initially we have a full text as black color.
    let attributedString = NSMutableAttributedString(string: text)
    // Extract mentions from metadata.
    let mentions = AmityMentionMapper.mentions(fromMetadata: metadata)
    // For each mention, we highlight part of the text.
    for mention in mentions {
        let range = NSRange(location: mention.index, length: mention.length + 1)
        if range.location != NSNotFound && range.location + range.length <= text.count {
            attributedString.addAttribute(.foregroundColor, value: UIColor.green, range: range)
        }
    }
    return attributedString
}

async function highlightMentionInText() {
  const rawText = 'Hey @alice, please check this out!'
  const mentionMetadata = [
    { userId: 'user-alice', displayName: 'alice', index: 4, length: 6 },
  ]

  let result = rawText
  mentionMetadata.slice().reverse().forEach(({ index, length, userId }) => {
    result =
      result.slice(0, index) +
      `<b data-user-id="${userId}">${result.slice(index, index + length)}</b>` +
      result.slice(index + length)
  })
  console.log('Highlighted:', result)
}
