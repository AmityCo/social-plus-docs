import { PostRepository } from '@amityco/ts-sdk'

function queryUserFeed(userId: string) {
  const query = PostRepository.getPosts({
    targetId: userId,
    targetType: 'user',
    sortBy: 'lastCreated',
  })
  const unsubscribe = query.on('dataUpdated', (posts) => {
    console.log('User feed posts:', posts)
  })
  return unsubscribe
}
