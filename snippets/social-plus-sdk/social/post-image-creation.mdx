func postImageCreation() {
        let feedRepository = AmityFeedRepository(client: AmityManager.shared.client!)
        let image = UIImage(named: "photo")!
        let imageData = image.jpegData(compressionQuality: 0.8)!
        
        let fileRepository = AmityFileRepository(client: AmityManager.shared.client!)
        fileRepository.uploadImage(imageData, fileName: "photo.jpg", mimeType: "image/jpeg") { uploadedImage, error in
            guard let imageId = uploadedImage?.fileId else { return }
            
            let builder = AmityImagePostBuilder()
            builder.setTargetType(.community)
            builder.setTargetId("communityId")
            builder.setImages([imageId])
            builder.setText("Check out this image!")
            
            feedRepository.createPost(builder) { post, error in
                if let error = error {
                    print("Image post creation failed: \(error)")
                    return
                }
                print("Image post created: \(post?.postId ?? "")")
            }
        }
    }
