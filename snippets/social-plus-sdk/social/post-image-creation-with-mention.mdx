import { PostRepository } from '@amityco/ts-sdk'

async function createImagePostWithMention() {
  const mentionedUsers = ['userId1', 'userId2'];
  const metadata = {
    mentioned: mentionedUsers.map(id => ({ type: 'user', userId: id })),
  };

  const { data: imageFile } = await PostRepository.uploadImage({ uri: 'file://path/to/image.jpg' });

  const postData = {
    targetId: 'communityId',
    targetType: 'community' as const,
    data: { text: 'Check this out @user1 @user2!' },
    metadata,
    mentionees: [{ type: 'user' as const, userIds: mentionedUsers }],
    attachments: [{ fileId: imageFile.fileId, type: 'image' as const }],
  };

  const livePost = PostRepository.createPost(postData);
  livePost.on('dataUpdated', post => console.log('Image post with mention created:', post));
}
