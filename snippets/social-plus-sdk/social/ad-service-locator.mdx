import { AdRepository } from '@amityco/ts-sdk'

function getNetworkAds() {
  const liveCollection = AdRepository.getNetworkAds()

  liveCollection.on('dataUpdated', (ads) => {
    console.log('Network ads updated:', ads.length)
    ads.forEach((ad) => {
      console.log('Ad ID:', ad.adId, '| Ad unit:', ad.adUnit)
    })
  })

  liveCollection.on('dataError', (error) => {
    console.error('Error fetching network ads:', error)
  })
}
