```typescript TypeScript
import { UserRepository } from '@amityco/ts-sdk'

function queryMyFollowerRequests(myUserId: string) {
  const liveCollection = UserRepository.getFollowers(myUserId, { status: 'pending' })

  liveCollection.on('dataUpdated', (pendingFollowers) => {
    console.log('Pending follow requests:', pendingFollowers.length)
    pendingFollowers.forEach((follower) => {
      console.log('Pending from:', follower.userId)
    })
  })

  liveCollection.on('dataError', (error) => {
    console.error('Error querying follow requests:', error)
  })

  return () => liveCollection.dispose()
}
```
