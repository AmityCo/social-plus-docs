import { StoryRepository } from '@amityco/ts-sdk'

function getStoryTargets(communityIds: string[]) {
  const targets = communityIds.map((targetId) => ({ targetId, targetType: 'community' as const }))

  const unsubscribe = StoryRepository.getTargetsByTargetIds(targets, ({ data: storyTargets, loading, error }) => {
    if (loading) return
    if (error) {
      console.error('Error fetching story targets:', error)
      return
    }

    storyTargets.forEach((target) => {
      console.log('Target:', target.targetId, '| Has stories:', target.hasStories)
    })
  })

  return unsubscribe
}

// Example usage
getStoryTargets(['communityId1', 'communityId2'])
