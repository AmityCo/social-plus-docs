func errorHandling() {
        let postRepository = AmityPostRepository(client: AmityManager.shared.client!)
        let builder = AmityTextPostBuilder()
        builder.setText("Test post")
        postRepository.createPost(builder, targetId: nil, targetType: .user) { post, error in
            if let amityError = error as? AmityError {
                switch amityError.code {
                case .unauthorized:
                    print("Unauthorized: \(amityError.localizedDescription)")
                case .itemNotFound:
                    print("Not found: \(amityError.localizedDescription)")
                default:
                    print("Error \(amityError.code.rawValue): \(amityError.localizedDescription)")
                }
                return
            }
            print("Post created: \(post?.postId ?? "")")
        }
    }
