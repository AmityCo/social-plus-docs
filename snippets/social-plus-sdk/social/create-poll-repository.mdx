import { PollRepository } from '@amityco/ts-sdk'

async function createPollRepository(question: string, choices: string[]) {
  try {
    const { data: poll } = await PollRepository.createPoll({
      question,
      answers: choices.map(text => ({ dataType: 'text', data: { text } })),
      timeToClosePoll: 3600,
      answerType: 'multiple',
    });
    console.log('Poll repository created poll:', poll.pollId);
    console.log('Answers count:', poll.answers?.length);
    return poll;
  } catch (error) {
    console.error('Poll repository operation failed:', error);
  }
}
