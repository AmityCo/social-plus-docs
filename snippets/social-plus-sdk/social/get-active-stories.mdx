import { StoryRepository } from '@amityco/ts-sdk'

function getActiveStories(targetId: string, targetType: string) {
  const unsubscribe = StoryRepository.getActiveStoriesByTarget(
    { targetId, targetType: targetType as 'community' },
    ({ data: stories, loading, error }) => {
      if (loading) return
      if (error) {
        console.error('Error fetching active stories:', error)
        return
      }

      console.log('Active stories:', stories.length)
      stories.forEach((story) => {
        console.log('Story ID:', story.storyId)
      })
    },
  )

  return unsubscribe
}

// Example usage
getActiveStories('communityId', 'community')
