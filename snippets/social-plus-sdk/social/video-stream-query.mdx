```typescript TypeScript
import { StreamRepository } from '@amityco/ts-sdk'

function queryStreams() {
  const liveCollection = StreamRepository.queryStreams({ status: ['live', 'recorded'] })

  liveCollection.on('dataUpdated', (streams) => {
    console.log('Streams found:', streams.length)
    streams.forEach((stream) => {
      console.log(' -', stream.streamId, '| status:', stream.status, '| title:', stream.title)
    })
  })

  liveCollection.on('dataError', (error) => {
    console.error('Error querying streams:', error)
  })
}
```
