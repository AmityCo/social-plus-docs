func postCreateFile() {
        let fileRepository = AmityFileRepository(client: AmityClient.shared)
        let fileData = AmityUploadableFile(fileURL: URL(fileURLWithPath: "/path/to/file.pdf"), fileName: "document.pdf")
        fileRepository.uploadFile(fileData, progress: nil) { fileData, error in
            guard let fileData = fileData, error == nil else {
                print("File upload failed: \(error!.localizedDescription)")
                return
            }
            let postRepository = AmityPostRepository(client: AmityClient.shared)
            let builder = AmityPostCreationBuilder()
            builder.setTargetType(.community, targetId: "communityId")
            builder.setFileIds([fileData.fileId])
            postRepository.createPost(builder) { post, error in
                print("File post created: \(post?.postId ?? "")")
            }
        }
    }
