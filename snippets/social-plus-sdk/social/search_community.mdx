void searchCommunity() async {
    final communityRepository = AmitySocialClient.newCommunityRepository();
    communityRepository
        .searchCommunity('keyword')
        .filter(AmityCommunityQueryFilter.ALL)
        .sortBy(AmityCommunitySortOption.DISPLAY_NAME)
        .asStream()
        .listen((communities) {
      print('Communities found: \${communities.length}');
    });
  }

async function searchCommunityUnderscore() {
