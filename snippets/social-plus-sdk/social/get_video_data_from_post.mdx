import { PostRepository, FileRepository } from '@amityco/ts-sdk'

async function getVideoDataFromPost() {
  const postId = 'video-post-id-123'
  const liveObject = PostRepository.getPost(postId)
  const unsubscribe = liveObject.subscribe(({ data: post, loading }) => {
    if (!loading && post && post.dataType === 'video') {
      const videoData = post.data as { fileId: string }
      const fileObject = FileRepository.getFile(videoData.fileId)
      fileObject.subscribe(({ data: file }) => {
        console.log('Video file URL:', file?.fileUrl)
      })
    }
  })
  return unsubscribe
}
