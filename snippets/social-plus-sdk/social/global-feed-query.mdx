```typescript TypeScript
import { FeedRepository } from '@amityco/ts-sdk'

function getGlobalFeed() {
  const liveCollection = FeedRepository.getGlobalFeed({})

  liveCollection.on('dataUpdated', (posts) => {
    console.log('Global feed posts:', posts.length)
    posts.forEach((post) => {
      console.log('Post ID:', post.postId)
    })
  })

  liveCollection.on('dataError', (error) => {
    console.error('Error fetching global feed:', error)
  })

  return () => liveCollection.dispose()
}
```
