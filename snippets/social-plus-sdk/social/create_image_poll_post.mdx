import { PostRepository, PollRepository } from '@amityco/ts-sdk'

async function createImagePollPost() {
  const poll = await PollRepository.createPoll({
    question: 'Which image do you prefer?',
    answers: [{ data: 'Option A' }, { data: 'Option B' }],
    closedIn: 86400000,
    dataType: 'single',
  })
  const post = await PostRepository.createPost({
    targetType: 'community',
    targetId: 'communityId',
    dataType: 'poll',
    data: { pollId: poll.pollId },
    attachments: [{ fileId: 'imageFileId', type: 'image' }],
  })
  console.log('Image poll post created:', post)
}
