function queryStoryTargets() {
  const disposer = StoryRepository.getStoryTargets(

    ({ data: targets, loading, error }) => {
      if (loading) return
      if (error) {
        console.error('Failed to query story targets', error)
        return
      }
      console.log('Story targets:', targets)
    },
