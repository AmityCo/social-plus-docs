import { CommunityRepository } from '@amityco/ts-sdk'

function getTrendingCommunities() {
  const disposer = CommunityRepository.getCommunities(
    { sortBy: 'lastCreated', isTrending: true, limit: 20 },
    ({ data: communities, loading, error }) => {
      if (loading) return
      if (error) {
        console.error('Failed to fetch trending communities', error)
        return
      }
      console.log('Trending communities:', communities)
    },
  )

  return disposer
}
