import { PostRepository } from '@amityco/ts-sdk'

function queryCommunityFeed(communityId: string) {
  const query = PostRepository.getPosts({
    targetId: communityId,
    targetType: 'community',
    sortBy: 'lastCreated',
  })
  const unsubscribe = query.on('dataUpdated', (posts) => {
    console.log('Community feed posts:', posts)
  })
  return unsubscribe
}
