        AmitySocialClient.newPostRepository()
            .createPost()
            .targetUser("userId")
            .image(listOf(imageUri))
            .build()
            .post()
            .subscribeOn(io.reactivex.rxjava3.schedulers.Schedulers.io())
            .observeOn(AndroidSchedulers.mainThread())
            .subscribe({ post ->
                val postId = post.getPostId()

            })

func createImagePost() {
        let postRepository = AmityPostRepository(client: AmityManager.shared.client!)
        let fileRepository = AmityFileRepository(client: AmityManager.shared.client!)
        let imageData = UIImage(named: "sample")!.jpegData(compressionQuality: 0.8)!
        fileRepository.uploadImage(imageData, progress: nil) { imageData, error in
            guard let imageData = imageData, error == nil else { return }
            let builder = AmityImagePostBuilder()
            builder.setImages([imageData])
            builder.setText("Check out this image!")
            postRepository.createPost(builder, targetId: nil, targetType: .user) { post, error in
                print("Image post created: \(post?.postId ?? "")")
            }
        }
    }
