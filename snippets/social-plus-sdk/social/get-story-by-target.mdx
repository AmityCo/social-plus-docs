import { StoryRepository } from '@amityco/ts-sdk'

function getStoriesByTarget() {
  const liveCollection = StoryRepository.getActiveStoriesByTarget({
    targetId: 'communityId',
    targetType: 'community',
  })

  liveCollection.on('dataUpdated', (stories) => {
    console.log('Active stories:', stories)
  })

  liveCollection.on('dataError', (error) => {
    console.error('Failed to get stories:', error)
  })

  return () => liveCollection.dispose()
}
