    // And iterate on each child post, to list out all available video qualities.
    func allAvailableVideoQuality(fromPost post: AmityPost) {
        let childrenPosts = post.childrenPosts
        for (index, childPost) in childrenPosts.enumerated() {
            print("Available videos qualities for video[\(index)]")
            for resolution in childPost.availableResolution() {
                let videoPath = childPost.getVideoInfo()?.getVideo(resolution: resolution) ?? "no-path"
                print("- resolution: \(resolution), path: \(videoPath)")
            }
        }
    }
