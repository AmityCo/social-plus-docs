import { CommunityRepository } from '@amityco/ts-sdk'

function observeCommunityMembership(communityId: string, userId: string) {
  const liveObject = CommunityRepository.Membership.getMember(
    communityId,
    userId,
    ({ data: member, loading, error }) => {
      if (error) { console.error(error); return; }
      if (!loading && member) {
        console.log('Membership status:', member.membership);
        console.log('Roles:', member.roles);
        console.log('Is banned:', member.isBanned);
      }
    }
  );
  return liveObject;
}
