            val isParentPost = post.getParentPostId() == null
            if(isParentPost) {
                // Text is available as a parent post
                val text = (post.getData() as AmityPost.Data.TEXT).getText()

                // Image, File, Video, Poll, and LiveStream are available as children
                val children = post.getChildren()
                children.forEach { childPost ->
                    getPostData(childPost)
                }
            } else {
                val data = post.getData()
                when(data) {
                    is AmityPost.Data.IMAGE -> {
                        val image: AmityImage? = data.getImage()
