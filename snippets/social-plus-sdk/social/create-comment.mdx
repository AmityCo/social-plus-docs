import { CommentRepository } from '@amityco/ts-sdk'

async function createComment(referenceId: string, text: string) {
  try {
    const { data: comment } = await CommentRepository.createComment({
      referenceId,
      referenceType: 'post',
      data: { text },
    });
    console.log('Comment created:', comment.commentId);
    return comment;
  } catch (error) {
    console.error('Failed to create comment:', error);
  }
}
