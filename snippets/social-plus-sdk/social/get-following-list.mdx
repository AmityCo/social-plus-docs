import { UserRepository } from '@amityco/ts-sdk'

function getFollowingList() {
  const liveCollection = UserRepository.getFollowings('userId')

  liveCollection.on('dataUpdated', (followings) => {
    console.log('Followings:', followings)
  })

  liveCollection.on('dataError', (error) => {
    console.error('Failed to get followings:', error)
  })

  return () => liveCollection.dispose()
}
