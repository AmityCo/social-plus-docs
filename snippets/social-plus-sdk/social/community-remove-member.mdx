fun communityRemoveMember(communityId: String, userId: String) {
        AmitySocialClient.newCommunityRepository()
            .membership(communityId = communityId)
            .removeMembers(userIds = listOf(userId))
            .subscribe(
                { println("Member removed: $userId") },
                { error -> println("Remove failed: ${error.message}") }
            )
    }
