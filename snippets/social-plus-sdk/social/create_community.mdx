func createCommunity() {
        let communityRepository = AmityCommunityRepository(client: AmityManager.shared.client!)
        let builder = AmityCommunityCreationDataBuilder()
        builder.setDisplayName("My New Community")
        builder.setDescription("A great community")
        builder.setIsPublic(true)
        communityRepository.createCommunity(with: builder) { community, error in
            if let error = error {
                print("Failed to create community: \(error)")
                return
            }
            print("Community created: \(community?.communityId ?? "")")
        }
    }
