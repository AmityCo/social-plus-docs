import { FeedRepository } from '@amityco/ts-sdk'

function getGlobalFeed() {
  const unsubscribe = FeedRepository.getGlobalFeed(
    {},
    (response) => {
      if (response.error) throw response.error
      if (!response.loading) {
        console.log('Global feed posts:', response.data)
      }
    },
  )
  return unsubscribe
}
