async function accessPostAndItsChildren(postId: string) {
  const liveObject = PostRepository.getPost(postId, ({ data: post, loading, error }) => {
    if (error) { console.error(error); return; }
    if (!loading && post) {
      console.log('Post:', post.postId, 'Children count:', post.children?.length);
      post.children?.forEach((childId: string) => {

        });
