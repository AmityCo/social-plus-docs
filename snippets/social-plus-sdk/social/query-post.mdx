    postRepository
        .getPosts()
        .targetUser('userId')
        .sortBy(AmityPostSortOption.LAST_CREATED)

import { PostRepository } from '@amityco/ts-sdk'

async function queryPost() {
  const params = {
    targetId: 'userId123',
    targetType: 'user' as const,
    sortBy: 'lastCreated' as const,
    feedType: 'published' as const,
  };

  const liveCollection = PostRepository.getPosts(params);
  liveCollection.on('dataUpdated', posts => {
    console.log('Posts:', posts.map(p => p.postId));
  });
  liveCollection.on('dataError', err => console.error('Error querying posts:', err));
}
