func feedGlobalFeedQuery() {
        let feedRepository = AmityFeedRepository(client: AmityManager.shared.client!)
        let liveCollection = feedRepository.getGlobalFeed()
        liveCollection.observe { collection, _, error in
            if let error = error { print("Error: \(error)"); return }
            let posts = collection.allObjects()
            print("Global feed loaded: \(posts.count) posts")
        }
