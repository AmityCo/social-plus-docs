import { UserRepository } from '@amityco/ts-sdk'

function getUserFollowInfo(userId: string) {
  const unsubscribe = UserRepository.Relationship.getFollowInfo(
    userId,
    ({ data: followInfo, loading, error }) => {
      if (loading) return
      if (error) throw error
      console.log('Follow status:', followInfo.status)
      console.log('Followers:', followInfo.followerCount)
      console.log('Following:', followInfo.followingCount)
    },
  )
  return unsubscribe
}
