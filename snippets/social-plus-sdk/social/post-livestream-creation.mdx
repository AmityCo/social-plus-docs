        postRepository: AmityPostRepository,

        description: String,
        thumbnailImage: AmityImage? = null,
        resolution: AmityBroadcastResolution = AmityBroadcastResolution.HD_720P,
        isSecure: Boolean = true,
        channelEnabled: Boolean? = true,
        targetType: AmityPost.TargetType,
        targetId: String,
    ) {
        AmityVideoClient.newStreamRepository()
            .createStream(
                title = title,
                description = description,
                thumbnailImage = thumbnailImage,
                resolution = resolution,
                isSecure = isSecure,
                channelEnabled = channelEnabled,
            )
            .flatMap { stream: AmityStream ->
                postRepository.createLiveStreamPost(
                    targetId = targetId,
                    targetType = targetType,
                    streamId = stream.getStreamId(),
                    text = "new video post!"
                )
            }
            .doOnSuccess { post: AmityPost ->
                // AmityPost
            }
            .doOnError {
                // Exception
            }
            .subscribe()
        }
