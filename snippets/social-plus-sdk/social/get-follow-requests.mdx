import { UserRepository } from '@amityco/ts-sdk'

function getPendingFollowRequests(userId: string) {
  const liveCollection = UserRepository.getFollowers(userId)

  liveCollection.on('dataUpdated', (followers) => {
    // Filter followers with 'pending' status to get follow requests
    const pendingRequests = followers.filter((follower) => follower.status === 'pending')
    console.log('Pending follow requests:', pendingRequests.length)
    pendingRequests.forEach((request) => {
      console.log(' -', request.userId)
    })
  })

  liveCollection.on('dataError', (error) => {
    console.error('Error fetching follow requests:', error)
  })
}
