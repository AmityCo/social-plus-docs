import { UserRepository } from '@amityco/ts-sdk'

function getMyFollowers(myUserId: string) {
  const liveCollection = UserRepository.getFollowers(myUserId)

  liveCollection.on('dataUpdated', (followers) => {
    console.log('My followers:', followers)
  })

  liveCollection.on('dataError', (error) => {
    console.error('Error fetching followers:', error)
  })

  return () => liveCollection.dispose()
}
