        let builder = AmityCommunityCreationDataBuilder()
        builder.setDisplayName("My Community")
        builder.setCommunityDescription("A community for Swift developers")
        builder.setIsPublic(true)
        builder.setCategoryIds(["categoryId1", "categoryId2"])
        
        communityRepository.createCommunity(with: builder) { community, error in
            if let error = error {
                print("Community creation failed: \(error)")
                return
            }
            print("Community created: \(community?.communityId ?? ""), displayName: \(community?.displayName ?? "")")
        }
    }
