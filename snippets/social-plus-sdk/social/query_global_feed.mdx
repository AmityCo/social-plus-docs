import { PostRepository } from '@amityco/ts-sdk'

function queryGlobalFeed() {
  const disposer = PostRepository.getGlobalFeed({ limit: 20 }, ({ data: posts, loading, error }) => {
    if (loading) return
    if (error) {
      console.error('Failed to query global feed', error)
      return
    }
    console.log('Global feed posts:', posts)
  })

  return disposer
}
