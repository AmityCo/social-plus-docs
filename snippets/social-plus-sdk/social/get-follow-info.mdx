import { UserRepository } from '@amityco/ts-sdk'

function getFollowInfo(userId: string) {
  const unsubscribe = UserRepository.getFollowInfo(userId, ({ data: followInfo, loading, error }) => {
    if (loading) return
    if (error) {
      console.error('Error fetching follow info:', error)
      return
    }

    console.log('Follow status:', followInfo.status)
    console.log('Follower count:', followInfo.followerCount)
    console.log('Following count:', followInfo.followingCount)
  })

  return unsubscribe
}
