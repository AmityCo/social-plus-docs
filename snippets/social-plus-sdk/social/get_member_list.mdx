func getMemberList() {
        let communityRepository = AmityCommunityRepository(client: AmityManager.shared.client!)
        let liveCollection = communityRepository.getMembers(communityId: "communityId", filter: .all, roles: [], sortBy: .firstCreated)
        liveCollection.observe { collection, _, error in
            if let error = error { print("Error: \(error)"); return }
            print("Member count: \(collection.count())")
        }
    }
