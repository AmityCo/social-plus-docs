func listCommunities() {
        let communityRepository = AmityCommunityRepository(client: AmityManager.shared.client!)
        let query = AmityCommunityQuery()
        query.filter = .all
        query.sortBy = .displayName
        let collection = communityRepository.getCommunities(with: query)
        collection.observe { collection, _, error in
            guard error == nil else { return }
            collection.snapshots.forEach { community in
                print("Community: \(community.displayName ?? "")")
            }
        }
    }
