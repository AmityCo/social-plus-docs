        let imageRepository = AmityImageRepository(client: AmityClient.shared)
        let image = UIImage(named: "sample")!
        imageRepository.uploadImage(image, progress: nil) { imageData, error in
            guard let imageData = imageData, error == nil else {
                print("Image upload failed: \(error!.localizedDescription)")
                return
            }
            let postRepository = AmityPostRepository(client: AmityClient.shared)
            let builder = AmityPostCreationBuilder()
            builder.setTargetType(.community, targetId: "communityId")
            builder.setImageIds([imageData.fileId])
            postRepository.createPost(builder) { post, error in
                print("Image post created: \(post?.postId ?? "")")
            }
        }
