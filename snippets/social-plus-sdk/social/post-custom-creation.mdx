func postCustomCreation() {
        let feedRepository = AmityFeedRepository(client: AmityManager.shared.client!)
        let builder = AmityPostCreationBuilder()
        builder.setText("Custom post text content")
        builder.setTargetType(.community)
        builder.setTargetId("communityId")
        let customData: [String: Any] = ["customKey": "customValue"]
        builder.setCustomData(customData)
        
        feedRepository.createPost(builder) { post, error in
            if let error = error {
                print("Custom post creation failed: \(error)")
                return
            }
            print("Custom post created: \(post?.postId ?? "")")
        }
    }
