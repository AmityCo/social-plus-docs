func postVideoCreation() {
        let feedRepository = AmityFeedRepository(client: AmityManager.shared.client!)
        let videoURL = URL(fileURLWithPath: "/path/to/video.mp4")
        let videoData = try! Data(contentsOf: videoURL)
        
        let fileRepository = AmityFileRepository(client: AmityManager.shared.client!)
        fileRepository.uploadVideo(videoData, fileName: "video.mp4", mimeType: "video/mp4") { uploadedVideo, error in
            guard let videoId = uploadedVideo?.fileId else { return }
            
            let builder = AmityVideoPostBuilder()
            builder.setTargetType(.community)
            builder.setTargetId("communityId")
            builder.setVideos([videoId])
            builder.setText("Check out this video!")
            
            feedRepository.createPost(builder) { post, error in
                if let error = error {
                    print("Video post creation failed: \(error)")
                    return
                }
                print("Video post created: \(post?.postId ?? "")")
            }
        }
    }
