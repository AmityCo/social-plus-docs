import { FeedRepository } from '@amityco/ts-sdk'

function getUserFeed(userId: string) {
  const unsubscribe = FeedRepository.getUserFeed(
    { userId, feedType: 'published' },
    ({ data: posts, loading, error }) => {
      if (loading) return
      if (error) {
        console.error('Error fetching user feed:', error)
        return
      }

      console.log(`Posts in feed for user ${userId}:`, posts.length)
      posts.forEach((post) => {
        console.log(' -', post.postId, '|', post.data?.text ?? post.dataType)
      })
    },
  )

  return unsubscribe
}
