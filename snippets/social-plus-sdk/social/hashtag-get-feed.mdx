import { PostRepository } from '@amityco/ts-sdk'

function getHashtagFeed(hashtag: string) {
  const posts = []

  const unsubscribe = PostRepository.searchPostsByHashtag(
    { hashtags: [hashtag] },
    (response) => {
      if (response.error) throw response.error
      if (!response.loading) {
        posts.length = 0
        posts.push(...response.data)
        console.log(`Posts with #${hashtag}:`, posts)
      }
    },
  )

  return unsubscribe
}
