import { StoryRepository } from '@amityco/ts-sdk'

function getStoryById(storyId: string) {
  const unsubscribe = StoryRepository.getStoryByStoryId(storyId, ({ data: story, loading, error }) => {
    if (loading) return
    if (error) {
      console.error('Error fetching story:', error)
      return
    }

    console.log('Story ID:', story.storyId)
    console.log('Story type:', story.dataType)
    console.log('Created at:', story.createdAt)
  })

  return unsubscribe
}
