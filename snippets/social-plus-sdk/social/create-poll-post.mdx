```typescript TypeScript
import { PostRepository } from '@amityco/ts-sdk'

async function createPollPost(targetId: string, question: string, answers: string[]) {
  try {
    const { data: poll } = await PollRepository.createPoll({
      question,
      answers: answers.map(text => ({ dataType: 'text', data: { text } })),
      timeToClosePoll: 86400,
      answerType: 'single',
    });
    const { data: post } = await PostRepository.createPost({
      targetType: 'community',
      targetId,
      dataType: 'poll',
      data: { pollId: poll.pollId },
    });
    console.log('Poll post created:', post.postId, 'pollId:', poll.pollId);
    return { post, poll };
  } catch (error) {
    console.error('Failed to create poll post:', error);
  }
}
```
