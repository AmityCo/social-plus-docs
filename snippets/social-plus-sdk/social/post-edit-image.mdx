import { PostRepository } from '@amityco/ts-sdk'

async function postEditImage(postId: string, text: string, imageFileIds: string[]) {
  const post = await PostRepository.updatePost({
    postId,
    dataType: 'image',
    data: { text },
    attachments: imageFileIds.map(fileId => ({ type: 'image', fileId })),
  })
  console.log('Image post edited:', post.postId)
  return post
}
