import { PostRepository } from '@amityco/ts-sdk'

function getGlobalPinnedPosts() {
  const liveCollection = PostRepository.getGlobalPinnedPosts({})

  liveCollection.on('dataUpdated', (posts) => {
    console.log('Pinned posts:', posts.length)
    posts.forEach((post) => {
      console.log('Post ID:', post.postId)
    })
  })

  })

  return () => liveCollection.dispose()
}
