async function accessPostInfo(postId: string) {
  const liveObject = PostRepository.getPost(postId, ({ data: post, loading, error }) => {
    if (error) { console.error(error); return; }
    if (!loading && post) {
      console.log('Post ID:', post.postId);
      console.log('Reach count:', post.reachCount);
      console.log('Impression count:', post.impressionCount);

  });
