fun getEvent(eventId: String): Disposable {
        val eventRepository = AmitySocialClient.newEventRepository()

        return eventRepository.getEvent(eventId)
            .subscribeOn(Schedulers.io())
            .observeOn(AndroidSchedulers.mainThread())
            .subscribe(
                { event: AmityEvent ->
                    // Event retrieved successfully
                    println("Event ID: ${event.getEventId()}")
                    println("Title: ${event.getTitle()}")
                    println("Description: ${event.getDescription()}")
                    println("Type: ${event.getType()}")
                    println("Start Time: ${event.getStartTime()}")
                    println("Location: ${event.getLocation()}")
                },
                { error: Throwable ->
                    // Handle error
                    println("Error retrieving event: ${error.message}")
                }
            )
    }
