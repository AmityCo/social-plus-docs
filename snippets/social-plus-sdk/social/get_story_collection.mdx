func getStoryCollection() {
        let storyRepository = AmityStoryRepository(client: AmityManager.shared.client!)
        let collection = storyRepository.getActiveStories(
            inTarget: AmityStoryTargetType.community,
            targetId: "communityId123",
            orderBy: .lastCreated
        )
        collection.observe { collection, _, error in
            guard error == nil else { return }
            print("Total stories: \(collection.count())")
        }
    }
