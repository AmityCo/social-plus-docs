import { PollRepository } from '@amityco/ts-sdk'

async function createPoll(question: string, answers: string[], closingInSeconds: number) {
  try {
    const { data: poll } = await PollRepository.createPoll({
      question,
      answers: answers.map(text => ({ dataType: 'text', data: { text } })),
      timeToClosePoll: closingInSeconds,
      answerType: 'single',
    });
    console.log('Poll created:', poll.pollId, 'Question:', poll.question);
    return poll;
  } catch (error) {
    console.error('Failed to create poll:', error);
  }
}
