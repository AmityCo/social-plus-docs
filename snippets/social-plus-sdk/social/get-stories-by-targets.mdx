import { StoryRepository } from '@amityco/ts-sdk'

function getStoriesByTargets(targetIds: string[]) {
  const targets = targetIds.map((targetId) => ({ targetId, targetType: 'community' as const }))

  const unsubscribe = StoryRepository.getStoriesByTargetIds(
    targets,
    ({ data: stories, loading, error }) => {
      if (loading) return
      if (error) {
        console.error('Error fetching stories by targets:', error)
        return
      }

      console.log('Stories:', stories.length)
      stories.forEach((story) => {
        console.log('Story ID:', story.storyId, '| Target:', story.targetId)
      })
    },
  )

  return unsubscribe
}

// Example usage
getStoriesByTargets(['communityId1', 'communityId2'])
