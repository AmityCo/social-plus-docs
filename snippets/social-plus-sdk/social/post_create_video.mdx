func postCreateVideo() {
        let videoRepository = AmityVideoRepository(client: AmityClient.shared)
        let videoURL = URL(fileURLWithPath: "/path/to/video.mp4")
        videoRepository.uploadVideo(videoURL, progress: nil) { videoData, error in
            guard let videoData = videoData, error == nil else { return }
            let postRepository = AmityPostRepository(client: AmityClient.shared)
            let builder = AmityPostCreationBuilder()
            builder.setTargetType(.community, targetId: "communityId")
            builder.setVideoIds([videoData.fileId])
            postRepository.createPost(builder) { post, error in
                print("Video post created: \(post?.postId ?? "")")
            }
        }
    }
