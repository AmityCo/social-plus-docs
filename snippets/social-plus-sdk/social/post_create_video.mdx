import { Client, createClient } from '@amityco/ts-sdk'

async function postCreateVideo() {
  const { PostRepository, uploadFile } = Amity;
  const videoFile = new File([''], 'video.mp4', { type: 'video/mp4' });
  const { data: fileData } = await uploadFile(videoFile, {
    onProgress: (pct: number) => console.log(`Upload progress: ${pct}%`),
  });
  const { data: post } = await PostRepository.createPost({
    targetId: 'communityId',
    targetType: 'community',
    dataType: 'video',
    data: { videoFileId: fileData.fileId },
  });
  console.log('Video post created:', post.postId);
}
