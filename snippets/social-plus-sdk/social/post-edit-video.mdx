import { PostRepository } from '@amityco/ts-sdk'

async function postEditVideo(postId: string, text: string, videoFileIds: string[]) {
  const post = await PostRepository.updatePost({
    postId,
    dataType: 'video',
    data: { text },
    attachments: videoFileIds.map(fileId => ({ type: 'video', fileId })),
  })
  console.log('Video post edited:', post.postId)
  return post
