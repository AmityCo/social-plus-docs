import { UserRepository } from '@amityco/ts-sdk'

function searchUsers(searchQuery: string) {
  const liveCollection = UserRepository.queryUsers({ displayName: searchQuery })

  liveCollection.on('dataUpdated', (users) => {
    console.log('User search results:', users)
  })

  liveCollection.on('dataError', (error) => {
    console.error('User search error:', error)
  })

  return () => liveCollection.dispose()
}
