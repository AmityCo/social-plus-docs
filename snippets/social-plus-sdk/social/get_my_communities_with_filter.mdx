func getMyCommunitiesWithFilter() {
        let communityRepository = AmityCommunityRepository(client: AmityManager.shared.client!)
        let queryOptions = AmityCommunityQueryOptions(filter: .member, sortBy: .displayName, includeDeleted: false)
        let liveCollection = communityRepository.getCommunities(with: queryOptions)
        liveCollection.observe { collection, _, error in
            if let error = error { print("Error: \(error)"); return }
            let communities = collection.allObjects()
            print("Found \(communities.count) communities")
        }
    }
