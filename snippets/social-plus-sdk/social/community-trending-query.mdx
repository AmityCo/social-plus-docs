import { CommunityRepository } from '@amityco/ts-sdk'

function getTrendingCommunities() {
  const liveCollection = CommunityRepository.getTrendingCommunities(
    { limit: 10 },
    ({ data: communities, loading, error }) => {
      if (error) { console.error(error); return; }
      if (!loading) {
        communities.forEach((c, i) => console.log(`#${i + 1}`, c.communityId, c.displayName));
      }
    }
  );
  return liveCollection;
}
