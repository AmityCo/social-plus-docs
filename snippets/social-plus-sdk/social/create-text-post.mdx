import { PostRepository } from '@amityco/ts-sdk'

async function createTextPost(userId: string, text: string) {
  const { data: post } = await PostRepository.createPost({
    targetId: userId,
    targetType: 'user',
    dataType: 'text',
    data: { text },
  })

  console.log('Text post created:', post.postId)
  return post
}

// Example usage
createTextPost('userId', 'Hello from flutter!')
