fun communityBanUser(communityId: String, userId: String) {
        AmitySocialClient.newCommunityRepository()
            .moderation(communityId = communityId)
            .banMembers(userIds = listOf(userId))
            .subscribe(
                { println("User $userId banned from $communityId") },
                { error -> println("Ban failed: ${error.message}") }
            )
    }
