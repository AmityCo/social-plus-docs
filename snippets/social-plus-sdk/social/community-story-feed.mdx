import { StoryRepository } from '@amityco/ts-sdk'

function getCommunityStoryFeed(communityId: string) {
  const liveCollection = StoryRepository.getActiveStoriesByTarget(
    { targetType: 'community', targetId: communityId },
    ({ data: stories, loading, error }) => {
      if (error) { console.error(error); return; }
      if (!loading) {
        stories.forEach(story => {
          console.log('Story:', story.storyId, 'type:', story.dataType);
        });
      }
    }
  );
  return liveCollection;
}
