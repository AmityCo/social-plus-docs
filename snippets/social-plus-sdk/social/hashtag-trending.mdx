async function hashtagTrending() {
  const liveCollection = PostRepository.getTrendingPosts({
    limit: 10,
  })
  const unsubscribe = liveCollection.subscribe(({ data: trending, loading }) => {
    if (!loading) {
      const hashtags = trending.flatMap(p => p.tags ?? [])
      const counts = hashtags.reduce((acc: Record<string, number>, tag) => {
        acc[tag] = (acc[tag] ?? 0) + 1
        return acc
      }, {})
      console.log('Trending hashtags:', counts)
    }
  })
  return unsubscribe
}
