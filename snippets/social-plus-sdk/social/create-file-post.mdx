import { PostRepository } from '@amityco/ts-sdk'

async function createFilePost(targetId: string, fileIds: string[]) {
  try {
    const { data: post } = await PostRepository.createPost({
      targetType: 'community',
      targetId,
      dataType: 'file',
      data: { fileId: fileIds[0] },
      attachments: fileIds.map(fileId => ({ fileId, type: 'file' })),
    });
    console.log('File post created:', post.postId);
    return post;
  } catch (error) {
    console.error('Failed to create file post:', error);
  }
}
