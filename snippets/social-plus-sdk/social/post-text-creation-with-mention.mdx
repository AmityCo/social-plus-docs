func postTextCreationWithMention() {
        let postRepository = AmityPostRepository(client: AmityClient.shared)
        let builder = AmityPostCreationBuilder()
        builder.setTargetType(.community, targetId: "communityId")
        builder.setText("Hello @user!")
        let mentionees = AmityMentioneesBuilder()
        mentionees.mentionUser(withId: "userId")
        builder.setMentionees(mentionees.build())
        postRepository.createPost(builder) { post, error in
            print("Text post with mention created: \(post?.postId ?? "")")
        }
    }

import { PostRepository } from '@amityco/ts-sdk'

async function createTextPostWithMention() {
  const mentionedUserIds = ['userId1', 'userId2'];

  const postData = {
    targetId: 'communityId',
    targetType: 'community' as const,
    data: { text: 'Hello @user1 and @user2!' },
    mentionees: [{ type: 'user' as const, userIds: mentionedUserIds }],
    metadata: {
      mentioned: mentionedUserIds.map(id => ({ type: 'user', userId: id })),
    },
  };

  const livePost = PostRepository.createPost(postData);
  livePost.on('dataUpdated', post => console.log('Text post with mention created:', post.postId));
}
