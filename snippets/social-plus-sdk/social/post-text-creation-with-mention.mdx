fun createTextPostWithMention(
        postRepository: AmityPostRepository
    ) {

        // To create a post on a community feed, create a community target
        val communityTarget = AmityPost.Target.COMMUNITY.create(communityId = "community1")

        // create AmityMentionMetadata from the creating text, using userId of the highlighting user
        // and startIndex and length of the highlighting area
        val text = "@userA writing text post"
        val userId = "userAId" // userId of userA
        val startMentionIndex = 0
        val mentionLength = 6

        val mentionMetadata: AmityMentionMetadata =
            AmityMentionMetadata.USER(
                userId = userId,
                index = startMentionIndex,
                length = mentionLength
            )

        // construct AmityMentionMetadata as JsonObject using AmityMentionMetadataCreator
        val mentionMetadataCreator: JsonObject =
            AmityMentionMetadataCreator(mentionMetaData = listOf(mentionMetadata)).create()

        // create a text post with mention
        postRepository.createTextPost(
                target = communityTarget,
                text = text,
                mentionUserIds = setOf(userId), // trigger a push notification to mentioned user
                metadata = mentionMetadataCreator //metadata to render mention highlights
            )
            .doOnSuccess { post: AmityPost ->
                // AmityPost
            }
            .doOnError {
                // Exception
            }
            .subscribe()
    }
