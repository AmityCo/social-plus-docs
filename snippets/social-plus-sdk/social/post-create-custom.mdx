func postCreateCustom() {
        let postRepository = AmityPostRepository(client: AmityClient.shared)
        let builder = AmityPostCreationBuilder()
        builder.setTargetType(.community, targetId: "communityId")
        builder.setText("Custom post")
        builder.setCustomData(["key": "value"])
        postRepository.createPost(builder) { post, error in
            if let error = error {
                print("Failed to create custom post: \(error.localizedDescription)")
                return
            }
            print("Custom post created: \(post?.postId ?? "")")
        }
    }
