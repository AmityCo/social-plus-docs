func getMyCommunityMembers() {
        let communityRepository = AmityCommunityRepository(client: AmityManager.shared.client!)
        let membershipRepository = communityRepository.membershipParticipation(forCommunityId: "communityId")
        
        let queryOptions = AmityCommunityMemberQueryOptions()
        queryOptions.roles = []
        let liveCollection = membershipRepository?.getMembers(options: queryOptions)
        
        let token = liveCollection?.observe { collection, changes, error in
            guard error == nil else { return }
            print("My community members count: \(collection.count())")
            for member in collection.allObjects() {
                print("Member: \(member.displayName ?? ""), userId: \(member.userId)")
            }
        }

    }
