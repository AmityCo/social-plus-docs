import { PostRepository } from '@amityco/ts-sdk';

async function flagPost(
  postId: Amity.Post['postId'],
  reasonReport: Amity.ContentFlagReason,
): Promise<boolean> {
  const isFlagged = await PostRepository.flagPost(postId, reasonReport);

  return isFlagged;
}
