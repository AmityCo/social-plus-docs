```dart Flutter
void removePostReaction(String postId) {
    // For Posts
    final reactionReference = AmityPostReactionReference(referenceId: postId);

    // For Comments
    final reactionReference = AmityCommentReactionReference(referenceId: commentId);

    // For Messages
    final reactionReference = AmityMessageReactionReference(referenceId: messageId);

    // For Stories
    final reactionReference = AmityStoryReactionReference(referenceId: storyId);
    
    AmitySocialClient.newReactionRepository()
        .removeReaction(reactionReference, 'like')
        .then((value) => {
              // Handle success
              print('Reaction removed successfully')
            })
        .onError((error) => {
              // Handle error
              print('Error removing reaction: $error')
            });
}
```
