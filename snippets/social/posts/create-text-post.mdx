```swift iOS
func createTextPost() async throws {
    let postRepository = AmityPostRepository(client: amityClient)

    let builder = AmityTextPostBuilder()
    builder.setText("Check out Premium Wool Socks!")

    // Define product tags in text with position info
    let productTags = [
        AmityTextProductTag(productId: "productId1", index: 10, length: 18)
    ]

    let post = try await postRepository.createTextPost(
        builder,
        targetId: "userId1",
        targetType: .user,
        metadata: nil,
        mentionees: nil,
        productTags: productTags
    )
}
```

```typescript TypeScript
import { PostRepository } from '@amityco/ts-sdk';

async function createTextPost(): Promise<Amity.Post> {
  const newPost = {
    targetType: "user",
    targetId: "userId1",
    data: {
      text: "Check out Premium Wool Socks!",
    },
    productTags: [
      {
        productId: "productId1",
        text: "Premium Wool Socks",
        index: 10,
        length: 18,
      },
    ],
  };

  const { data: post } = await PostRepository.createPost(newPost);
  return post;
}
```
