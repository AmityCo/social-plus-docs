```dart Flutter
void createTextPost() {
  AmitySocialClient.newPostRepository()
      .createPost()
      .targetUser('userId')
      .text('Check out Premium Wool Socks!')
      .post()
      .then((AmityPost post) {
        // Success - text post created
        print('Text post created successfully');
      })
      .onError((error, stackTrace) {
        // Handle error
        print('Error creating text post: $error');
      });
}
```
