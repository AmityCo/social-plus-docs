```swift iOS
Task { @MainActor in
    do {
        let options = AmityEventCreateOptions(
            title: "Community Workshop 2025",
            description: "Join us for an interactive workshop on iOS development",
            type: .virtual,
            startTime: Date(),
            endTime: Date(),
            originType: .community,
            originId: "community-123",
            metadata: ["timezone": "America/Los_Angeles"]
        )
        
        let event = try await AmityEventRepository.shared.createEvent(options: options)
        print("Event created: \(event.eventId)")
    } catch {
        print("Failed to create event: \(error)")
    }
}
```

```kotlin Android
AmityEventRepository.createEvent(
    options = AmityEventCreateOptions()
        .setTitle("Community Workshop 2025")
        .setDescription("Join us for an interactive workshop on Android development")
        .setType(AmityEventType.VIRTUAL)
        .setStartTime("2025-03-20T14:00:00Z")
        .setOriginType(AmityEventOriginType.COMMUNITY)
        .setOriginId("community-123")
        .setMetadata(mapOf("timezone" to "America/Los_Angeles"))
).subscribe(
    onSuccess = { event ->
        Log.d("Event", "Created event: ${event.eventId}")
    },
    onError = { error ->
        Log.e("Event", "Failed to create event", error)
    }
)
```

```typescript TypeScript
import { EventRepository, AmityEventType, AmityEventOriginType } from '@amityco/ts-sdk';

try {
    const { data: event } = await EventRepository.createEvent({
        title: "Community Workshop 2025",
        description: "Join us for an interactive workshop on web development",
        type: AmityEventType.Virtual,
        startTime: "2025-03-20T14:00:00Z",
        endTime: "2025-03-20T16:00:00Z",
        originType: AmityEventOriginType.Community,
        originId: "community-123",
        metadata: {
            timezone: "America/Los_Angeles"
        }
    });
    console.log("Event created:", event.eventId);
} catch (error) {
    console.error("Failed to create event:", error);
}
```
