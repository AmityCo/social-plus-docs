import { notificationTray } from '@amityco/ts-sdk';

const unsubscribe = notificationTray.getNotificationTrayItems(
  { limit: 20 },
  ({ data: items, onNextPage, hasNextPage, loading, error }) => {
    if (loading) return;
    if (error) {
      handleError(error);
      return;
    }

    renderResults(items);

    if (hasNextPage) {
      onNextPage?.();
    }
  },
);

unsubscribe();
