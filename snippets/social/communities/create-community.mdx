fun createCommunity(displayName: String, isPublic: Boolean) {
        AmitySocialClient.newCommunityRepository()
            .createCommunity(displayName = displayName)
            .isPublic(isPublic = isPublic)
            .build()
            .create()
            .doOnSuccess { community: AmityCommunity ->
                // handle created community
            }
            .doOnError { /* handle error */ }
            .subscribe()
    }
