class MyFeedViewController: UIViewController, UITableViewDataSource {
    
    var feedCollection: AmityCollection<AmityPost>!
    var token: AmityNotificationToken?
    
    @IBOutlet var tableView: UITableView!
    
    override func viewDidLoad() {
        super.viewDidLoad()
        tableView.register(UITableViewCell.self, forCellReuseIdentifier: "cell")
        tableView.dataSource = self
        setupLiveCollection()
    }
    
    func setupLiveCollection() {
        // Note: In this example, we assume `feedRepository` is created and stored in the app-wide scope.
        feedCollection = App.shared.feedRepository.getMyFeedSorted(by: .lastCreated, includeDeleted: false)
        token = feedCollection.observe { [weak self] collection, changes, error in
            self?.tableView.reloadData()
        }
    }
    
    @IBAction func loadNextButtonDidTouch() {
        // Note: In this example, we assume to have `loadNext` button in this page.
        // When the user tap at the button, it will trigger live collection
        // To load the next page, so the user can see more post in the feed.
        if feedCollection.hasNext {
            feedCollection.nextPage()
        }
    }
    
    func numberOfSections(in tableView: UITableView) -> Int {
        1
    }
    
    func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        Int(feedCollection.count())
    }
    
    func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        let cell = tableView.dequeueReusableCell(withIdentifier: "cell")!
        if let post = feedCollection.object(at: indexPath.row) {
            cell.textLabel?.text = post.data?["text"] as? String
        } else {
            cell.textLabel?.text = "unknown"
        }
        return cell
    }
    
}
