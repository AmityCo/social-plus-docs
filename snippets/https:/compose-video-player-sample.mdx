@Composable
    @androidx.media3.common.util.UnstableApi
    fun PlayLiveStream(
        stream: AmityStream

        AmityComposeVideoPlayer { amityVideoPlayer ->
            amityVideoPlayer.play(streamId = stream.getStreamId())
        }
    }

    @Composable
    @androidx.media3.common.util.UnstableApi
    fun AmityComposeVideoPlayer(
        modifier: Modifier = Modifier,
        callback: (AmityVideoPlayer) -> Unit
    ) {
        AndroidView(
            factory = { context: Context ->
                AmityVideoPlayer(context = context)
            },
            update = { amityVideoPlayer ->
                callback(amityVideoPlayer)
            },
            modifier = modifier.fillMaxSize()
        )
    }
