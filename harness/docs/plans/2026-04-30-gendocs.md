# `gendocs` Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a `gendocs` command to the Go harness that reads SDK snippet files from all 4 platforms and generates/updates Mintlify `<CodeGroup>` MDX files under `social-plus-docs/snippets/`.

**Architecture:** The existing `scanner.Scan()` already returns `Snippet{Content, AscPage, Filename, Platform}` — `Content` is clean code with markers stripped. The new `internal/docgen` package groups snippets by a key derived from `Filename`, derives an MDX output path from `asc_page`, renders `<CodeGroup>` MDX, and writes files idempotently. A `cmd/gendocs.go` entry point wires it all together.

**Tech Stack:** Go 1.23, existing `internal/scanner`, `internal/config` packages, `encoding/json`, `strings`, `path/filepath`, `os`.

```go
package docgen

import (
	"path"
	"strings"
	"unicode"
)

// DeriveKey converts a snippet filename to a URL-safe grouping key.
// "AmityCommunityMembershipAddMembers.kt" → "community-membership-add-members"
func DeriveKey(filename string) string {
	base := filename
	if idx := strings.LastIndex(base, "."); idx >= 0 {
		base = base[:idx]
	}
	if strings.HasPrefix(strings.ToLower(base), "amity") {
		base = base[5:]
	}
	return pascalToKebab(base)
}

// DeriveMDXPath returns the relative MDX output path for a snippet group.
// Uses the first two segments of ascPage as directory + key as filename.
// "social-plus-sdk/social/communities/..." + "community-create" → "social-plus-sdk/social/community-create.mdx"
func DeriveMDXPath(ascPage, key string) string {
	segs := strings.SplitN(ascPage, "/", 3)
	dir1 := safeSegment(segs, 0)
	dir2 := safeSegment(segs, 1)
	return path.Join(dir1, dir2, key+".mdx")
}

func safeSegment(segs []string, i int) string {
	if i < len(segs) {
		return segs[i]
	}
	return "unknown"
}

func pascalToKebab(s string) string {
	var b strings.Builder
	for i, r := range s {
		if unicode.IsUpper(r) && i > 0 {
			b.WriteRune('-')
		}
		b.WriteRune(unicode.ToLower(r))
	}
	return b.String()
}
```

```go
import "social-plus/harness/internal/scanner"

// SnippetGroup holds all platform snippets for one logical function.
type SnippetGroup struct {
	Key         string
	AscPage     string
	Description string
	Snippets    map[string]scanner.Snippet // platform → snippet
}

// GroupSnippets groups a flat list of snippets by their derived key.
// Snippets with the same base filename (across platforms) are merged into one group.
func GroupSnippets(snips []scanner.Snippet) map[string]SnippetGroup {
	groups := make(map[string]SnippetGroup)
	for _, s := range snips {
		key := DeriveKey(s.Filename)
		g, exists := groups[key]
		if !exists {
			g = SnippetGroup{
				Key:         key,
				AscPage:     s.AscPage,
				Description: s.Description,
				Snippets:    make(map[string]scanner.Snippet),
			}
		}
		g.Snippets[s.Platform] = s
		groups[key] = g
	}
	return groups
}
```

```go
import "fmt"

var platformOrder = []string{"android", "ios", "flutter", "typescript"}

var platformLabel = map[string]string{
	"android":    "Android",
	"ios":        "iOS",
	"flutter":    "Flutter",
	"typescript": "TypeScript",
}

var platformFence = map[string]string{
	"android":    "kotlin",
	"ios":        "swift",
	"flutter":    "dart",
	"typescript": "typescript",
}

// RenderMDX generates the MDX content for a snippet group.
// date is used in the generated header comment (format: "YYYY-MM-DD").
func RenderMDX(group SnippetGroup, date string) string {
	var b strings.Builder
	fmt.Fprintf(&b, "{/* AUTO-GENERATED by harness gendocs — do not edit manually */}\n")
	fmt.Fprintf(&b, "{/* function: %s | updated: %s */}\n\n", group.Key, date)
	b.WriteString("<CodeGroup>\n")
	for _, platform := range platformOrder {
		snip, ok := group.Snippets[platform]
		if !ok {
			continue
		}
		fence := platformFence[platform]
		label := platformLabel[platform]
		fmt.Fprintf(&b, "\n```%s %s\n%s\n```\n", fence, label, snip.Content)
	}
	b.WriteString("\n</CodeGroup>\n")
	return b.String()
}
```

```go
import (
	"encoding/json"
	"os"
	"path/filepath"
	"sort"
	"time"
)

const autoGeneratedMarker = "AUTO-GENERATED by harness gendocs"

// Write generates MDX files for all groups under outDir.
// Returns counts of files written and skipped (human-edited).
// In dry-run mode, counts planned writes but touches no files.
func Write(outDir string, groups map[string]SnippetGroup, dryRun bool) (written, skipped int, err error) {
	date := time.Now().Format("2006-01-02")
	for _, group := range groups {
		relPath := DeriveMDXPath(group.AscPage, group.Key)
		absPath := filepath.Join(outDir, filepath.FromSlash(relPath))

		if existing, readErr := os.ReadFile(absPath); readErr == nil {
			if !strings.Contains(string(existing), autoGeneratedMarker) {
				fmt.Printf("[gendocs] SKIP (human-edited): %s\n", relPath)
				skipped++
				continue
			}
		}

		content := RenderMDX(group, date)
		fmt.Printf("[gendocs] WRITE: %s\n", relPath)
		written++

		if dryRun {
			continue
		}

		if mkErr := os.MkdirAll(filepath.Dir(absPath), 0755); mkErr != nil {
			return written, skipped, fmt.Errorf("mkdir %s: %w", filepath.Dir(absPath), mkErr)
		}
		if writeErr := os.WriteFile(absPath, []byte(content), 0644); writeErr != nil {
			return written, skipped, fmt.Errorf("write %s: %w", absPath, writeErr)
		}
	}
	return written, skipped, nil
}

// WriteManifest writes manifest.json mapping function key → MDX relative path.
func WriteManifest(outDir string, groups map[string]SnippetGroup, dryRun bool) error {
	manifest := make(map[string]string, len(groups))
	for key, group := range groups {
		manifest[key] = DeriveMDXPath(group.AscPage, key)
	}

	keys := make([]string, 0, len(manifest))
	for k := range manifest {
		keys = append(keys, k)
	}
	sort.Strings(keys)
	sorted := make(map[string]string, len(manifest))
	for _, k := range keys {
		sorted[k] = manifest[k]
	}

	b, err := json.MarshalIndent(sorted, "", "  ")
	if err != nil {
		return fmt.Errorf("marshal manifest: %w", err)
	}

	manifestPath := filepath.Join(outDir, "manifest.json")
	fmt.Printf("[gendocs] WRITE manifest: %s\n", manifestPath)
	if dryRun {
		return nil
	}
	if mkErr := os.MkdirAll(outDir, 0755); mkErr != nil {
		return fmt.Errorf("mkdir %s: %w", outDir, mkErr)
	}
	return os.WriteFile(manifestPath, b, 0644)
}
```

```go
package main

import (
	"flag"
	"fmt"
	"os"
	"path/filepath"

	"social-plus/harness/internal/config"
	"social-plus/harness/internal/docgen"
	"social-plus/harness/internal/scanner"
)

func runGendocs(args []string) {
	fs := flag.NewFlagSet("gendocs", flag.ExitOnError)
	cfgPath := fs.String("config", "harness-config.yml", "path to harness-config.yml")
	dryRun := fs.Bool("dry-run", false, "print planned writes without touching files")
	outOverride := fs.String("out", "", "override output directory (default: <docs_path>/snippets)")
	if err := fs.Parse(args); err != nil {
		fmt.Fprintf(os.Stderr, "parse flags: %v\n", err)
		os.Exit(1)
	}

	cfg, err := config.Load(*cfgPath)
	if err != nil {
		fmt.Fprintf(os.Stderr, "load config: %v\n", err)
		os.Exit(1)
	}

	var allSnips []scanner.Snippet
	for platform, sdkCfg := range cfg.SDKs {
		sdkBase := filepath.Join(filepath.Dir(*cfgPath), sdkCfg.Path)
		snippetDir := filepath.Join(sdkBase, sdkCfg.SnippetDir)
		snips, scanErr := scanner.Scan(snippetDir, platform)
		if scanErr != nil {
			fmt.Fprintf(os.Stderr, "[%s] scan error: %v\n", platform, scanErr)
			os.Exit(1)
		}
		allSnips = append(allSnips, snips...)
	}

	groups := docgen.GroupSnippets(allSnips)
	fmt.Printf("[gendocs] %d snippets → %d groups\n", len(allSnips), len(groups))

	outDir := *outOverride
	if outDir == "" {
		outDir = filepath.Join(filepath.Dir(*cfgPath), cfg.Docs.Path, "snippets")
	}
	if *dryRun {
		fmt.Println("[gendocs] DRY RUN — no files will be written")
	}

	written, skipped, err := docgen.Write(outDir, groups, *dryRun)
	if err != nil {
		fmt.Fprintf(os.Stderr, "write error: %v\n", err)
		os.Exit(1)
	}
	if err := docgen.WriteManifest(outDir, groups, *dryRun); err != nil {
		fmt.Fprintf(os.Stderr, "manifest error: %v\n", err)
		os.Exit(1)
	}

	fmt.Printf("[gendocs] done — written: %d, skipped (human-edited): %d\n", written, skipped)
}
```
