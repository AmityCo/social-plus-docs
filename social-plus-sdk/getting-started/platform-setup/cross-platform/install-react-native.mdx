# React Native SDK Installation

The Social+ TypeScript SDK works natively with React Native applications, providing full social functionality.

## React Native Compatibility

The Social+ TypeScript SDK works natively with React Native applications. For most modern React Native projects, you can use the SDK out of the box. 

<Info>
If you need to support older JavaScript runtimes, add a polyfill for `Object.fromEntries`:

1. Install core-js:
```bash
npm install core-js
```
2. Import the polyfill at the root of your project:
```javascript
import 'core-js/features/object/from-entries';
```
</Info>

## Requirements

- React Native 0.60+
- Node.js 14+
- iOS 13.0+ / Android 5.0 (API 21)+

## Platform Setup

<Tabs>
<Tab title="iOS">

### CocoaPods Configuration

Add to your `ios/Podfile`:

```ruby
platform :ios, '13.0'

target 'YourApp' do
  use_react_native!
  
  # Add any additional pods here
end
```

Install pods:

```bash
cd ios && pod install
```

### Info.plist Configuration

Add network permissions to `ios/YourApp/Info.plist`:

```xml
<key>NSAppTransportSecurity</key>
<dict>
  <key>NSAllowsArbitraryLoads</key>
  <true/>
</dict>
```
</Tab>

<Tab title="Android">

### Gradle Configuration

Ensure your `android/build.gradle` has the correct settings:

```gradle
buildscript {
    ext {
        buildToolsVersion = "30.0.3"
        minSdkVersion = 21
        compileSdkVersion = 30
        targetSdkVersion = 30
    }
}
```

### Permissions

Add to `android/app/src/main/AndroidManifest.xml`:

```xml
<uses-permission android:name="android.permission.INTERNET" />
<uses-permission android:name="android.permission.ACCESS_NETWORK_STATE" />
```

### ProGuard Rules (if using)

Add to `android/app/proguard-rules.pro`:

```proguard
# Social+ SDK
-keep class co.amity.** { *; }
-dontwarn co.amity.**
```

## Initialize the SDK

Create a client instance in your app:

```tsx
import React, { useEffect, useState } from 'react';
import { Client } from '@amityco/ts-sdk';

const App = () => {
  const [client, setClient] = useState<Client | null>(null);

  useEffect(() => {
    const initializeSDK = async () => {
      try {
        const amityClient = Client.create({
          apiKey: 'your-api-key',
          region: 'sg'
        });

        await amityClient.login({
          userId: 'user123',
          displayName: 'John Doe',
          authToken: 'your-auth-token'
        });

        setClient(amityClient);
      } catch (error) {
        console.error('SDK initialization failed:', error);
      }
    };

    initializeSDK();
  }, []);

  if (!client) {
    return <LoadingScreen />;
  }

  return <MainApp client={client} />;
};

export default App;
```

## Context Provider Pattern

Create a context for app-wide SDK access:

```tsx
import React, { createContext, useContext, useEffect, useState } from 'react';
import { Client } from '@amityco/ts-sdk';

interface AmityContextType {
  client: Client | null;
  isReady: boolean;
  login: (userId: string, displayName: string) => Promise<void>;
  logout: () => Promise<void>;
}

const AmityContext = createContext<AmityContextType | undefined>(undefined);

export const AmityProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [client, setClient] = useState<Client | null>(null);
  const [isReady, setIsReady] = useState(false);

  useEffect(() => {
    const initClient = async () => {
      const amityClient = Client.create({
        apiKey: 'your-api-key',
        region: 'sg'
      });
      
      setClient(amityClient);
      setIsReady(true);
    };

    initClient();
  }, []);

  const login = async (userId: string, displayName: string) => {
    if (!client) throw new Error('Client not initialized');
    
    await client.login({
      userId,
      displayName,
      authToken: 'your-auth-token'
    });
  };

  const logout = async () => {
    if (!client) return;
    await client.logout();
  };

  return (
    <AmityContext.Provider value={{ client, isReady, login, logout }}>
      {children}
    </AmityContext.Provider>
  );
};

export const useAmity = () => {
  const context = useContext(AmityContext);
  if (!context) {
    throw new Error('useAmity must be used within AmityProvider');
  }
  return context;
};
```

## Using the SDK in Components

```tsx
import React, { useEffect, useState } from 'react';
import { View, Text, FlatList } from 'react-native';
import { useAmity } from './AmityContext';

const CommunityList = () => {
  const { client, isReady } = useAmity();
  const [communities, setCommunities] = useState([]);

  useEffect(() => {
    if (!isReady || !client) return;

    const fetchCommunities = async () => {
      try {
        const response = await client.getCommunities();
        setCommunities(response.data);
      } catch (error) {
        console.error('Failed to fetch communities:', error);
      }
    };

    fetchCommunities();
  }, [client, isReady]);

  const renderCommunity = ({ item }) => (
    <View style={{ padding: 16 }}>
      <Text style={{ fontSize: 18, fontWeight: 'bold' }}>
        {item.displayName}
      </Text>
      <Text>{item.description}</Text>
    </View>
  );

  return (
    <FlatList
      data={communities}
      renderItem={renderCommunity}
      keyExtractor={(item) => item.communityId}
    />
  );
};

export default CommunityList;
```

## Error Handling

Implement comprehensive error handling:

```tsx
import { Alert } from 'react-native';

const handleAmityError = (error: any) => {
  let message = 'An error occurred';
  
  switch (error.code) {
    case 'INVALID_API_KEY':
      message = 'Invalid API key. Please check your configuration.';
      break;
    case 'NETWORK_ERROR':
      message = 'Network connection failed. Please check your internet connection.';
      break;
    case 'UNAUTHORIZED':
      message = 'Authentication failed. Please log in again.';
      break;
    default:
      message = error.message || 'An unexpected error occurred';
  }
  
  Alert.alert('Error', message);
};

// Usage in components
try {
  await client.login({ userId: 'user123', displayName: 'John Doe' });
} catch (error) {
  handleAmityError(error);
}
```

## State Management Integration

<Tabs>
<Tab title="Redux Toolkit">
```tsx
import { createSlice, createAsyncThunk } from '@reduxjs/toolkit';
import { Client } from '@amityco/ts-sdk';

export const initializeAmity = createAsyncThunk(
  'amity/initialize',
  async (apiKey: string) => {
    const client = Client.create({
      apiKey,
      region: 'sg'
    });
    return client;
  }
);

const amitySlice = createSlice({
  name: 'amity',
  initialState: {
    client: null,
    isReady: false,
    error: null
  },
  reducers: {},
  extraReducers: (builder) => {
    builder
      .addCase(initializeAmity.fulfilled, (state, action) => {
        state.client = action.payload;
        state.isReady = true;
      })
      .addCase(initializeAmity.rejected, (state, action) => {
        state.error = action.error.message;
      });
  }
});

export default amitySlice.reducer;
```
</Tab>

<Tab title="Zustand">
```tsx
import { create } from 'zustand';
import { Client } from '@amityco/ts-sdk';

interface AmityStore {
  client: Client | null;
  isReady: boolean;
  initialize: (apiKey: string) => Promise<void>;
  login: (userId: string, displayName: string) => Promise<void>;
  logout: () => Promise<void>;
}

export const useAmityStore = create<AmityStore>((set, get) => ({
  client: null,
  isReady: false,
  
  initialize: async (apiKey: string) => {
    const client = Client.create({
      apiKey,
      region: 'sg'
    });
    
    set({ client, isReady: true });
  },
  
  login: async (userId: string, displayName: string) => {
    const { client } = get();
    if (!client) throw new Error('Client not initialized');
    
    await client.login({ userId, displayName });
  },
  
  logout: async () => {
    const { client } = get();
    if (client) {
      await client.logout();
    }
  }
}));
```
